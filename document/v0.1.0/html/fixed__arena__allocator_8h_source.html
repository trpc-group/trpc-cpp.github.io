37class FixedMemoryPool {
38 public:
40 explicit FixedMemoryPool(size_t item_num, size_t item_mem_size) noexcept : item_num_(item_num) {
41 item_mem_size_ = std::max(item_mem_size, sizeof(size_t));
42 size_t mod_size = item_mem_size_ % sizeof(size_t);
43 item_mem_size_ = (mod_size > 0) ? (item_mem_size_ - mod_size + sizeof(size_t)) : item_mem_size_;
44
45 void* mem_ptr = ::aligned_alloc(64, item_mem_size_ * item_num_);
46 assert(mem_ptr != nullptr);
47
48 unsigned char* ptr = static_cast<unsigned char*>(mem_ptr);
49 for (uint32_t i = 0; i != item_num_; ptr += item_mem_size_) {
50 ++i;
51 memcpy(ptr, &i, sizeof(uint32_t));
52 }
53
54 mem_ptr_ = static_cast<unsigned char*>(mem_ptr);
55 avail_head_index_ = 0;
56 avail_num_ = item_num_;
57 }
58
60 ~FixedMemoryPool() {
61 assert(mem_ptr_ != nullptr);
62 ::free(static_cast<void*>(mem_ptr_));
63 }
64
66 bool Allocate(void** mem_ptr) noexcept {
67 if (avail_num_ == 0) {
68 return false;
69 }
70
71 unsigned char* ptr = mem_ptr_ + (avail_head_index_ * item_mem_size_);
72
73 --avail_num_;
74
75 avail_head_index_ = *(reinterpret_cast<uint32_t *>(ptr));
76
77 memset(ptr, 0, item_mem_size_);
78
79 *mem_ptr = ptr;
80
81 return true;
82 }
83
85 bool Deallocate(void* mem_ptr) noexcept {
86 unsigned char* prelease = static_cast<unsigned char*>(mem_ptr);
87 if (prelease < mem_ptr_ || prelease > (mem_ptr_ + item_mem_size_ * item_num_)) {
88 // Not allocated from current pool.
89 return false;
90 }
91
92 if (avail_num_ >= item_num_) {
93 assert(false);
94 return false;
95 }
96
97 assert((prelease - mem_ptr_) % item_mem_size_ == 0);
98
99 *(reinterpret_cast<uint32_t *>(prelease)) = avail_head_index_;
100
101 avail_head_index_ = static_cast<uint32_t>((prelease - mem_ptr_) / item_mem_size_);
102
103 ++avail_num_;
104
105 return true;
106 }
107
108 private:
109 FixedMemoryPool() = delete;
110
111 private:
112 unsigned char* mem_ptr_{nullptr};
113
114 uint32_t item_num_{0};
115
116 uint32_t avail_head_index_{0};
117
118 uint32_t avail_num_{0};
119
120 uint32_t item_mem_size_{0};
121};
