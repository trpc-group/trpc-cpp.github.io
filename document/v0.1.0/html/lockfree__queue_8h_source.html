34class alignas(64) LockFreeQueue {
35 public:
36 enum {
37 RT_OK = 0,
38 RT_FULL = 1,
39 RT_EMPTY = 2,
40 RT_ERR = -1,

52 int Init(size_t size) {
53 if (init_) {
54 return RT_OK;
55 }
56
57 bool size_is_power_of_2 = (size >= 2) && ((size & (size - 1)) == 0);
58 if (!size_is_power_of_2) {
59 uint64_t tmp = 1;
60 while (tmp <= size) {
61 tmp <<= 1;
62 }
63
64 size = tmp;
65 }
66
67 mask_ = size - 1;
68 capacity_ = size;
69
70 elements_ = std::make_unique<Element[]>(capacity_);
71 for (size_t i = 0; i < size; ++i) {
72 elements_[i].sequence = i;
73 }
74
75 count_.store(0, std::memory_order_relaxed);
76
77 enqueue_pos_.store(0, std::memory_order_relaxed);
78 dequeue_pos_.store(0, std::memory_order_relaxed);
79
80 return RT_OK;
81 }

86 int Enqueue(const T& val) {
87 Element* elem = nullptr;
88 size_t pos = enqueue_pos_.load(std::memory_order_relaxed);
89
90 while (true) {
91 elem = &elements_[pos & mask_];
92 size_t seq = elem->sequence.load(std::memory_order_acquire);
93 intptr_t dif = (intptr_t)seq - (intptr_t)pos;
94
95 if (dif == 0) {
96 if (enqueue_pos_.compare_exchange_weak(pos, pos + 1, std::memory_order_relaxed)) {
97 break;
98 }
99 } else if (dif < 0) {
100 if (pos - dequeue_pos_.load(std::memory_order_relaxed) == capacity_) return RT_FULL; // queue full
101
102 pos = enqueue_pos_.load(std::memory_order_relaxed);
103 } else {
104 pos = enqueue_pos_.load(std::memory_order_relaxed);
105 }
106 }
107
108 elem->data = val;
109 elem->sequence.store(pos + 1, std::memory_order_release);
110
111 count_.fetch_add(1, std::memory_order_release);
112
113 return RT_OK;
114 }
115
119 int Dequeue(T& val) {
120 Element* elem = nullptr;
121 size_t pos = dequeue_pos_.load(std::memory_order_relaxed);
122
123 while (true) {
124 elem = &elements_[pos & mask_];
125 size_t seq = elem->sequence.load(std::memory_order_acquire);
126 intptr_t dif = (intptr_t)seq - (intptr_t)(pos + 1);
127
128 if (dif == 0) {
129 if (dequeue_pos_.compare_exchange_weak(pos, pos + 1, std::memory_order_relaxed)) {
130 break;
131 }
132 } else if (dif < 0) {
133 if (pos - enqueue_pos_.load(std::memory_order_relaxed) == 0) return RT_EMPTY; // queue empty
134
135 pos = dequeue_pos_.load(std::memory_order_relaxed);
136 } else {
137 pos = dequeue_pos_.load(std::memory_order_relaxed);
138 }
139 }
140
141 val = elem->data;
142 elem->sequence.store(pos + mask_ + 1, std::memory_order_release);
143
144 count_.fetch_sub(1, std::memory_order_release);
145
146 return RT_OK;
147 }

151 uint32_t Size() const { return count_.load(std::memory_order_relaxed); }
152
155 uint32_t Capacity() const { return static_cast<uint32_t>(capacity_); }
