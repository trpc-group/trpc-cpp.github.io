98 template <typename Sink, typename SinkConfig>
99 bool InitSink(const char* logger_name) {
100 auto& instance = instances_[logger_name];
101 auto& conf = instance.config;
102 // Get local_file configuration
103 SinkConfig sink_config;
104 // configuration does not exist and returns true
105 if (!GetLoggerConfig<SinkConfig>(conf.name, sink_config)) return true;
106 auto sink = MakeRefCounted<Sink>();
107 sink->Init(sink_config);
108 // Add the new sink to the logger's sinks
109 instance.logger->sinks().push_back(sink->SpdSink());
110
111 return true;
112 }
