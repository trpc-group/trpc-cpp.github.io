34template <class RequestType, class ResponseType>
35class StreamRpcMethodHandler : public RpcMethodHandlerInterface {
36 public:
37 using ServerStreamFunction =
38 std::function<Status(const ServerContextPtr&, const RequestType&, stream::StreamWriter<ResponseType>*)>;
39
40 using ClientStreamFunction =
41 std::function<Status(const ServerContextPtr&, const stream::StreamReader<RequestType>&, ResponseType*)>;
42
43 using BidiStreamFunction = std::function<Status(const ServerContextPtr, const stream::StreamReader<RequestType>&,
44 stream::StreamWriter<ResponseType>*)>;
45
46 explicit StreamRpcMethodHandler(const ServerStreamFunction& func)
47 : server_stream_func_(func), type_(MethodType::SERVER_STREAMING) {}
48
49 explicit StreamRpcMethodHandler(const ClientStreamFunction& func)
50 : client_stream_func_(func), type_(MethodType::CLIENT_STREAMING) {}
51
52 explicit StreamRpcMethodHandler(const BidiStreamFunction& func)
53 : bidi_stream_func_(func), type_(MethodType::BIDI_STREAMING) {}
54
55 ~StreamRpcMethodHandler() override {}
56
57 void Execute(const ServerContextPtr& context, NoncontiguousBuffer&& req_data,
58 NoncontiguousBuffer& rsp_data) noexcept override {
59 TRPC_ASSERT(false && "Unreachable");
60 }
61
62 void Execute(const ServerContextPtr& context) noexcept override {
63 using R = RequestType;
64 using W = ResponseType;
65
66 // StreamReaderWriterProvider object will be moved from context.
67 auto stream_provider = context->GetStreamReaderWriterProvider();
68 Status status{kSuccStatus};
69
70 // Starts the stream, and waits for the stream to be ready.
71 status = stream_provider->Start();
72 if (!status.OK()) {
73 stream_provider->Reset(status);
74 return;
75 }
76
77 // Initializes codec options.
78 stream::MessageContentCodecOptions content_codec_ops{
79 .serialization = serialization::SerializationFactory::GetInstance()->Get(context->GetReqEncodeType()),
80 .content_type = context->GetReqEncodeType(),
81 };
82
83 auto rw = stream::Create<W, R>(stream_provider, content_codec_ops);
84
85 // Dispatches streaming RPC method.
86 switch (type_) {
87 case MethodType::SERVER_STREAMING: {
88 R request{};
89 // Waits for the request, default timeout: 32s.
90 // FIXME: Reads timeout from configure.
91// int read_timeout = context->GetService()->GetServiceAdapterOption().stream_read_timeout;
92 int read_timeout = 32000;
93 status = rw.Read(&request, read_timeout);
94 if (status.OK()) {
95 auto writer = rw.Writer();
96 status = server_stream_func_(context, request, &writer);
97 }
98 break;
99 }
100 case MethodType::CLIENT_STREAMING: {
101 W response{};
102 auto reader = rw.Reader();
103 status = client_stream_func_(context, reader, &response);
104 Status write_status = rw.Write(response);
105 if (!write_status.OK()) {
106 status = write_status;
107 }
108 break;
109 }
110 case MethodType::BIDI_STREAMING: {
111 auto reader = rw.Reader();
112 auto writer = rw.Writer();
113 status = bidi_stream_func_(context, reader, &writer);
114 break;
115 }
116 default: {
117 TRPC_LOG_ERROR("unsupported method");
118 status = Status{-1, -1, "unsupported method"};
119 }
120 }
121
122 // Sets final status of RPC execution.
123 context->SetStatus(status);
124 stream_provider->Close(status);
125 }
126
127 private:
128 // Server-side streaming RPC.
129 ServerStreamFunction server_stream_func_;
130 // Client-side streaming RPC.
131 ClientStreamFunction client_stream_func_;
132 // Bidirectional streaming RPC.
133 BidiStreamFunction bidi_stream_func_;
134 trpc::MethodType type_;
135};
