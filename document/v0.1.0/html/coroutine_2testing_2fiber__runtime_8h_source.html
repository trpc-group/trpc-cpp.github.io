28template <class F>
29void RunAsFiber(F&& f) {
30 trpc::fiber::StartRuntime();
31 trpc::runtime::SetRuntimeType(trpc::runtime::kFiberRuntime);
32 trpc::Latch l(1);
33 trpc::Fiber([&, f = std::forward<F>(f)] {
34 f();
35 l.count_down();
36 }).Detach();
37 l.wait();
38 trpc::fiber::TerminateRuntime();
