47class Semaphore {
48 private:
49 struct entry {
50 Promise<> pr;
51 size_t num;
52 trpc::internal::SinglyLinkedListEntry chain;
53 entry(Promise<>&& p, size_t n) noexcept : pr(std::move(p)), num(n) {}
54 };
55
56 trpc::internal::SinglyLinkedList<entry, &entry::chain> wait_list_;
57
58 size_t count_;
59
60 Exception* ex_ = nullptr;
61
62 int32_t worker_id_;
63
64 int32_t doing_;
65
66 bool HasAvailableUnits(size_t num) const noexcept {
67 return count_ >= 0 && count_ >= num;
68 }
69
70 bool MayProceed(size_t num) const noexcept {
71 return HasAvailableUnits(num) && wait_list_.empty();
72 }
73
74 int32_t GetThreadId() const noexcept {
75 if (TRPC_UNLIKELY(nullptr == WorkerThread::GetCurrentWorkerThread())) {
76 // If not in trpc thread model, does not check for thread safety, not recommended.
77 return -1;
78 }
79 return static_cast<int32_t>(WorkerThread::GetCurrentWorkerThread()->Id());
80 }
81
82 public:
83 static constexpr size_t MaxCounter() noexcept {
84 return std::numeric_limits<decltype(count_)>::max();
85 }
86
87 explicit Semaphore(size_t count) noexcept : count_(count), doing_(0) {
88 worker_id_ = GetThreadId();
89 }
90
91 ~Semaphore() noexcept {
92 if (ex_) {
93 delete ex_;
94 }
95 }
96
104 Future<> Wait(size_t num = 1) noexcept {
105 if (TRPC_UNLIKELY(worker_id_ != GetThreadId())) {
106 return MakeExceptionFuture<>(
107 SemaphoreException("Not on the same worker thread",
108 SemaphoreException::NOT_THREAD_SAFE));
109 }
110
111 // No need to wait
112 if (MayProceed(num)) {
113 count_ -= num;
114 return MakeReadyFuture<>();
115 }
116
117 // It's broken
118 if (ex_) {
119 return MakeExceptionFuture<>(*ex_);
120 }
121
122 try {
123 auto e = new entry(Promise<>(), num);
124 wait_list_.push_back(e);
125 return e->pr.GetFuture();
126 } catch (...) {
127 return MakeExceptionFuture<>(
128 SemaphoreException("Failed to add promise to list",
129 SemaphoreException::LIST_ADD));
130 }
131 }
132
139 bool Signal(size_t num = 1) noexcept {
140 if (TRPC_UNLIKELY(worker_id_ != GetThreadId())) {
141 return false;
142 }
143
144 if (ex_) {
145 return false;
146 }
147
148 count_ += num;
149
150 if (doing_ > 0) {
151 // some promises has not SetValue, DO NOT pop new waiter.
152 return true;
153 }
154
155 decltype(wait_list_) ready_list;
156
157 // FIFO, current count can meet the request, make them ready.
158 while (!wait_list_.empty() && HasAvailableUnits(wait_list_.front().num)) {
159 auto e = wait_list_.pop_front();
160 count_ -= e->num;
161 ready_list.push_back(e);
162 }
163
164 doing_ = ready_list.size();
165 while (!ready_list.empty()) {
166 auto e = ready_list.pop_front();
167 doing_--;
168 e->pr.SetValue();
169 delete e;
170 }
171
172 return true;
173 }
174
179 bool TryWait(size_t num = 1) noexcept {
180 if (TRPC_UNLIKELY(worker_id_ != GetThreadId())) {
181 return false;
182 }
183
184 if (MayProceed(num)) {
185 count_ -= num;
186 return true;
187 }
188 return false;
189 }
190
193 // @param msg will be used to construct exception, default to empty.
194 void Broke(const char* msg = nullptr) noexcept {
195 const char* m = "Broke the semaphore";
196 if (msg) {
197 m = msg;
198 }
199 ex_ = new Exception(SemaphoreException(m, SemaphoreException::BROKEN));
200 count_ = 0;
201 while (!wait_list_.empty()) {
202 auto& e = wait_list_.front();
203 e.pr.SetException(*ex_);
204 delete wait_list_.pop_front();
205 }
206 return;
207 }
208
209 size_t AvailableUnits() const noexcept { return count_; }
210
211 size_t Waiters() const noexcept { return wait_list_.size(); }
212
213 uint32_t WorkerId() const noexcept { return worker_id_; }
214};
