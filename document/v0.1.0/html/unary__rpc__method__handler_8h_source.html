74 bool PreExecute(const ServerContextPtr& context, NoncontiguousBuffer&& req_body) {
75 TRPC_ASSERT(context->GetServerCodec() != nullptr);
76
77 CreateReqAndRspObj(context);
78
79 auto decompress_type = context->GetReqCompressType();
80 auto ret = compressor::DecompressIfNeeded(decompress_type, req_body);
81 if (TRPC_UNLIKELY(!ret)) {
82 std::string err_msg = "request body decompress failed, decompress type:";
83 err_msg += std::to_string(decompress_type);
84 SetErrorStatus(context, codec::ServerRetCode::DECODE_ERROR, std::move(err_msg));
85 return false;
86 }
87
88 uint8_t encode_type = context->GetReqEncodeType();
89 auto serialization = serialization::SerializationFactory::GetInstance()->Get(encode_type);
90 if (serialization == nullptr) {
91 std::string error_msg = "request body deserialize failed, not support encode_type:";
92 error_msg += std::to_string(encode_type);
93 SetErrorStatus(context, codec::ServerRetCode::DECODE_ERROR, std::move(error_msg));
94 return false;
95 }
96
97 ret = Deserialize(serialization, &req_body, context->GetRequestData());
98 if (TRPC_UNLIKELY(!ret)) {
99 std::string error_msg = "request body deserialize failed, encode_type:";
100 error_msg += std::to_string(encode_type);
101 SetErrorStatus(context, codec::ServerRetCode::DECODE_ERROR, std::move(error_msg));
102 return false;
103 }
104
105 req_body.Clear();
106
107 auto& filter_controller = context->GetFilterController();
108 auto filter_status = filter_controller.RunMessageServerFilters(FilterPoint::SERVER_PRE_RPC_INVOKE, context);
109
110 return filter_status != FilterStatus::REJECT;
111 }
112
113 void PostExecute(const ServerContextPtr& context, NoncontiguousBuffer& rsp_body) {
114 TRPC_CHECK(context->IsResponse());
115
116 auto& filter_controller = context->GetFilterController();
117 filter_controller.RunMessageServerFilters(FilterPoint::SERVER_POST_RPC_INVOKE, context);
118
119 if (TRPC_UNLIKELY(!context->GetStatus().OK())) {
120 return;
121 }
122
123 uint8_t encode_type = context->GetRspEncodeType();
124 auto serialization = serialization::SerializationFactory::GetInstance()->Get(encode_type);
125 bool ret = Serialize(serialization, context->GetResponseData(), rsp_body);
126 if (TRPC_UNLIKELY(!ret)) {
127 std::string error_msg = "response body serialize failed, encode_type:";
128 error_msg += std::to_string(encode_type);
129 SetErrorStatus(context, codec::ServerRetCode::ENCODE_ERROR, std::move(error_msg));
130 return;
131 }
132
133 ret = compressor::CompressIfNeeded(context->GetRspCompressType(), rsp_body, context->GetRspCompressLevel());
134 if (TRPC_UNLIKELY(!ret)) {
135 std::string error_msg = "response body serialize compress, compress_type:";
136 error_msg += std::to_string(context->GetRspCompressType());
137 SetErrorStatus(context, codec::ServerRetCode::ENCODE_ERROR, "response body compress failed");
138 return;
139 }
140
141 DestroyReqAndRspObj(context);
142 }

160 void CreateReqObj(const ServerContextPtr& context) {

201 bool Serialize(serialization::Serialization* serialization, void* rsp, NoncontiguousBuffer& buff) {
202 serialization::DataType type;
203 if constexpr (std::is_convertible_v<RequestType*, google::protobuf::Message*>) {
204 type = serialization::kPbMessage;
205 } else if constexpr (std::is_convertible_v<RequestType*, rapidjson::Document*>) {
206 type = serialization::kRapidJson;
207 } else if constexpr (std::is_convertible_v<RequestType*, flatbuffers::trpc::MessageFbs*>) {
208 type = serialization::kFlatBuffers;
209 } else if constexpr (std::is_convertible_v<RequestType*, NoncontiguousBuffer*>) {
210 type = serialization::kNonContiguousBufferNoop;
211 } else if constexpr (std::is_convertible_v<RequestType*, std::string*>) {
212 type = serialization::kStringNoop;
213 } else {
214 return false;
215 }
216
217 return serialization->Serialize(type, rsp, &buff);
218 }

220 bool Deserialize(serialization::Serialization* serialization, NoncontiguousBuffer* req_data, void* req) {
221 serialization::DataType type;
222 if constexpr (std::is_convertible_v<RequestType*, google::protobuf::Message*>) {
223 type = serialization::kPbMessage;
224 } else if constexpr (std::is_convertible_v<RequestType*, rapidjson::Document*>) {
225 type = serialization::kRapidJson;
226 } else if constexpr (std::is_convertible_v<RequestType*, flatbuffers::trpc::MessageFbs*>) {
227 type = serialization::kFlatBuffers;
228 } else if constexpr (std::is_convertible_v<RequestType*, NoncontiguousBuffer*>) {
229 type = serialization::kNonContiguousBufferNoop;
230 } else if constexpr (std::is_convertible_v<RequestType*, std::string*>) {
231 type = serialization::kStringNoop;
232 } else {
233 return false;
234 }
235
236 return serialization->Deserialize(req_data, type, req);
237 }
