21template <typename T>
22class ThreadLockT {
23 public:
24 // Lock during construction.
25 explicit ThreadLockT(const T& mutex) : mutex_(mutex) {
26 mutex_.Lock();
27 acquired_ = true;
28 }
29
30 // Unlock during destruction.
31 virtual ~ThreadLockT() {
32 if (acquired_) {
33 mutex_.Unlock();
34 }
35 }
36
37 // Not implemented; prevents accidental use.
38 ThreadLockT(const ThreadLockT&) = delete;
39 ThreadLockT& operator=(const ThreadLockT&) = delete;
40
43 bool Acquire() const {
44 if (acquired_) {
45 return false;
46 }
47 mutex_.Lock();
48 acquired_ = true;
49 return true;
50 }
51
54 bool TryAcquire() const {
55 acquired_ = mutex_.TryLock();
56 return acquired_;
57 }
58
61 bool Release() const {
62 if (!acquired_) {
63 return false;
64 }
65 mutex_.Unlock();
66 acquired_ = false;
67 }
68
71 bool Acquired() const { return acquired_; }
72
73 protected:
74 // Used for lock attempt operations, similar to ThreadLockT.
75 ThreadLockT(const T& mutex, bool) : mutex_(mutex) { acquired_ = mutex_.TryLock(); }
76
77 protected:
78 // Lock object.
79 const T& mutex_;
80
81 // Whether the lock has been acquired.
82 mutable bool acquired_;
83};
