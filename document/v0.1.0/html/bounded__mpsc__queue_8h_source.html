31class alignas(64) BoundedMPSCQueue {
32 public:

36 BoundedMPSCQueue() = default;

40 bool Init(size_t size) {
41 capacity_ = RoundUpPowerOf2(size);
42
43 elements_ = std::make_unique<Element[]>(capacity_);
44 for (size_t i = 0; i < capacity_; ++i) {
45 elements_[i].sequence = i;
46 }
47
48 enqueue_pos_.store(0, std::memory_order_relaxed);
49 dequeue_pos_.store(0, std::memory_order_relaxed);
50
51 return true;
52 }

57 bool Push(T data) {
58 size_t pos = 0;
59 Element* elem;
60
61 while (true) {
62 pos = enqueue_pos_.load(std::memory_order_relaxed);
63 elem = &elements_[pos & (capacity_ - 1)];
64 size_t seq = elem->sequence.load(std::memory_order_acquire);
65 intptr_t dif = (intptr_t)seq - (intptr_t)pos;
66
67 if (dif == 0) {
68 if (enqueue_pos_.compare_exchange_weak(pos, pos + 1, std::memory_order_relaxed)) {
69 break;
70 }
71 } else if (dif < 0) {
72 if (pos - dequeue_pos_.load(std::memory_order_relaxed) == capacity_) {
73 return false;
74 }
75 }
76
77 queue::detail::Pause();
78 }
79
80 elem->data = std::move(data);
81 elem->sequence.store(pos + 1, std::memory_order_release);
82
83 return true;
84 }

128 // template <class = std::enable_if_t<std::is_trivial_v<T>>>
129 bool Dequeue(T* data) {
130 size_t pos = dequeue_pos_.load(std::memory_order_relaxed);
131 Element* elem = &elements_[pos & (capacity_ - 1)];
132 size_t seq = elem->sequence.load(std::memory_order_acquire);
133 intptr_t dif = (intptr_t)seq - (intptr_t)(pos + 1);
134
135 if (dif == 0) {
136 dequeue_pos_.store(pos + 1, std::memory_order_relaxed);
137 *data = std::move(elem->data);
138 elem->sequence.store(pos + capacity_, std::memory_order_release);
139
140 return true;
141 }
142
143 return false;
144 }

147 size_t Size() const {
148 size_t enqueue_pos = enqueue_pos_.load(std::memory_order_relaxed);
149 size_t dequeue_pos = dequeue_pos_.load(std::memory_order_relaxed);
150 return (enqueue_pos <= dequeue_pos ? 0 : (enqueue_pos - dequeue_pos));
151 }
