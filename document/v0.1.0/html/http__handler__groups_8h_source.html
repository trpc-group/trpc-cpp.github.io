146 template <typename T, MethodType Method>
147 HttpHandlerGroups& Handle(std::string_view path) {
148 auto iter = handlers_->find(typeid(T));
149 if (iter == handlers_->end()) {

181 template <MethodType Method>
182 HttpHandlerGroups& Handle(std::string_view path, std::shared_ptr<HandlerBase> handler) {
183 routes_.Add(Method, HttpPath(NormalizePath(path)), std::move(handler));
184 return *this;

187 std::string NormalizePath(std::string_view path) const {
188 if (path.empty()) {
189 return basic_path_;
190 }
191 std::string full_path = basic_path_;
192 if (path[0] != '/') {
193 full_path.push_back('/');
194 }
195 if (path[path.length() - 1] == '/') {
196 path.remove_suffix(1);
197 }
198 return full_path.append(path);
199 }
200
201 static trpc::http::Path HttpPath(std::string path) {
202 if (path.find('<') != std::string::npos && path.find('>') != std::string::npos) {
203 path.insert(0, "<ph(").append(")>");
204 }
205 return trpc::http::Path(std::move(path));
206 }

208 private:
209 Routes& routes_;
210 std::shared_ptr<std::unordered_map<std::type_index, std::shared_ptr<HandlerBase>>> handlers_;
211 std::string basic_path_;
