47template <typename T>
48class PassiveStatus {
49 public:
51 using FucType = std::function<T()>;
53 using SamplerOp = OpAdd<T>;
56 using SamplerInvOp = OpMinus<T>;
58 using ReducerSamplerType = ReducerSampler<PassiveStatus, T, SamplerOp, SamplerInvOp>;
60 using SeriesSamplerType = SeriesSampler<PassiveStatus, T, SamplerOp>;
61
62 public:
65 explicit PassiveStatus(FucType const& fn) : fn_(fn) {}
66
70 explicit PassiveStatus(std::string_view rel_path, FucType const& fn) : fn_(fn) {
71 handle_ = LinkToParent(rel_path, TrpcVarGroup::FindOrCreate("/"));
72 }
73
78 explicit PassiveStatus(TrpcVarGroup* parent, std::string_view rel_path, FucType const& fn)
79 : fn_(fn) {
80 handle_ = LinkToParent(rel_path, parent);
81 }
82
84 ~PassiveStatus() {
85 if (reducer_sampler_) {
86 reducer_sampler_->Destroy();
87 }
88 if (series_sampler_) {
89 series_sampler_->Destroy();
90 }
91 }
92
95 std::weak_ptr<ReducerSamplerType> GetReducerSampler() {
96 if (!reducer_sampler_) {
97 reducer_sampler_ = std::make_shared<ReducerSamplerType>(this);
98 // Here notify to sampler collector.
99 reducer_sampler_->Schedule();
100 }
101 return reducer_sampler_;
102 }
103
105 T GetValue() const noexcept { return (fn_ ? fn_() : T()); }
106
109 T Reset() {
110 TRPC_ASSERT(false && "InternalPassiveStatus::Reset() should never be called, abort");
111 }
112
114 std::string GetAbsPath() const { return IsExposed() ? handle_->abs_path : std::string(); }
115
117 bool IsExposed() const { return handle_.has_value(); }
118
120 std::weak_ptr<SeriesSamplerType> GetSeriesSampler() { return series_sampler_; }
121
123 std::unordered_map<std::string, std::vector<T>> GetSeriesValue() const noexcept {
124 if (series_sampler_) {
125 return series_sampler_->GetSeries();
126 }
127 return std::unordered_map<std::string, std::vector<T>>();
128 }
129
130 private:
131 // Check series sampler is enabled or not, if not and allowed, then create it.
132 // Not thread-safe.
133 bool CheckSeries() {
134 // Not all T is allowed to open series sampler.
135 if constexpr (!std::is_integral_v<T> && !std::is_floating_point_v<T>) {
136 return false;
137 }
138
139 // Config is required too.
140 if (!trpc::TrpcConfig::GetInstance()->GetGlobalConfig().tvar_config.save_series) {
141 return false;
142 }
143
144 // Not enable and allowed, just create it.
145 if (!series_sampler_) {
146 series_sampler_ = std::make_shared<SeriesSamplerType>(this);
147 // Notify to sampler collector.
148 series_sampler_->Schedule();
149 }
150 return true;
151 }
152
153 // Exposed tvar will try to open series sampler.
154 // If series sampler is not enabled, history data is not available in admin.
155 std::optional<TrpcVarGroup::Handle> LinkToParent(std::string_view rel_path,
156 TrpcVarGroup* parent) {
157 if (CheckSeries()) {
158 return TrpcVarGroup::LinkToParent(
159 rel_path, parent, [this] { return detail::ToJsonValue(GetValue()); },
160 [this] { return detail::ToJsonValue(GetSeriesValue()); });
161 } else {
162 return TrpcVarGroup::LinkToParent(rel_path, parent,
163 [this] { return detail::ToJsonValue(GetValue()); });
164 }
165 }
166
167 private:
168 // User defined function to return current value.
169 FucType fn_;
170 // To implement window data, to support Window/PerSecond.
171 std::shared_ptr<ReducerSamplerType> reducer_sampler_;
172 // To implement history data.
173 std::shared_ptr<SeriesSamplerType> series_sampler_;
174 // Must be last member.
175 std::optional<TrpcVarGroup::Handle> handle_;
176};
