35class HttpStreamHandler : public StreamHandlerBase {
36 public:
37 TRPC_HTTP_STREAM_HANDLER_METHOD(Get)
38 TRPC_HTTP_STREAM_HANDLER_METHOD(Head)
39 TRPC_HTTP_STREAM_HANDLER_METHOD(Post)
40 TRPC_HTTP_STREAM_HANDLER_METHOD(Put)
41 TRPC_HTTP_STREAM_HANDLER_METHOD(Delete)
42 TRPC_HTTP_STREAM_HANDLER_METHOD(Options)
43 TRPC_HTTP_STREAM_HANDLER_METHOD(Patch)
44
45 virtual Status Handle(const ServerContextPtr& ctx, const RequestPtr& req, Response* rsp) {
46 Status status;
47 try {
48 switch (req->GetMethodType()) {
49 TRPC_HTTP_STREAM_HANDLER_CASE(GET, Get);
50 TRPC_HTTP_STREAM_HANDLER_CASE(HEAD, Head);
51 TRPC_HTTP_STREAM_HANDLER_CASE(POST, Post);
52 TRPC_HTTP_STREAM_HANDLER_CASE(PUT, Put);
53 TRPC_HTTP_STREAM_HANDLER_CASE(DELETE, Delete);
54 TRPC_HTTP_STREAM_HANDLER_CASE(OPTIONS, Options);
55 TRPC_HTTP_STREAM_HANDLER_CASE(PATCH, Patch);
56 default:
57 rsp->SetStatus(Response::StatusCode::kNotImplemented);
58 }
59 } catch (const BaseException& http_ex) {
60 rsp->SetStatus(http_ex.Status(), http_ex.str());
61 } catch (...) { // do not return other exception messages to users for safety reasons
62 rsp->SetStatus(Response::StatusCode::kInternalServerError);
63 }
64 rsp->Done();
65 return status;
66 }
67
72 Status HandleStream(const std::string&, ServerContextPtr ctx, RequestPtr req, Response* rsp) final {
73 return Handle(ctx, req, rsp);
74 }
75
76 using StreamHandlerBase::Handle;
77};
