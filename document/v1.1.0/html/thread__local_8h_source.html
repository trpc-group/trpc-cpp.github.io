26// Same as `ThreadLocal<T>` except that objects are initialized eagerly (also
27// nondeterministically.). Note that `T`'s constructor may not touch other TLS
28// variables, otherwise the behavior is undefined.
29//
30// Performs slightly better. For internal use only.
31//
32// Instances of `T` in different threads are guaranteed to reside in different
33// cache line. However, if `T` itself allocates memory, there's no guarantee on
34// how memory referred by `T` in different threads are allocated.
35//
36template <class T>
37class ThreadLocalAlwaysInitialized {
38 public:
39 ThreadLocalAlwaysInitialized();
40 ~ThreadLocalAlwaysInitialized();
41
42 // Initialize object with a customized initializer.
43 template <class F>
44 explicit ThreadLocalAlwaysInitialized(F&& initializer);
45
46 // Noncopyable / nonmovable.
47 ThreadLocalAlwaysInitialized(const ThreadLocalAlwaysInitialized&) = delete;
48 ThreadLocalAlwaysInitialized& operator=(const ThreadLocalAlwaysInitialized&) = delete;
49
50 // Accessors.
51 T* Get() const;
52 T* operator->() const { return Get(); }
53 T& operator*() const { return *Get(); }
54
55 // Traverse through all instances among threads.
56 //
57 // CAUTION: Called with internal lock held. You may not touch TLS in `f`.
58 // (Maybe we should name it as `ForEachUnsafe` or `ForEachLocked`?)
59 template <class F>
60 void ForEach(F&& f) const;
61
62 private:
63 // Placed as the first member to keep accessing it quick.
64 //
65 // Always a multiple of `kEntrySize`.
66 std::ptrdiff_t offset_;
67 Function<void(void*)> initializer_;
68};

72// Performance note:
73//
74// In some versions of Google's gperftools (tcmalloc), allocating memory from
75// different threads often results in adjacent addresses (within a cacheline
76// boundary). This allocation scheme CAN EASILY LEAD TO FALSE-SHARING AND HURT
77// PERFORMANCE. As we often use `ThreadLocal<T>` for perf. optimization, this (I
78// would say it's a bug) totally defeat the reason why we want a "thread-local"
79// copy in the first place.
80//
81// Due to technical reasons (we don't have control on how instances of `T` are
82// allocated), we can't workaround this quirk for you automatically, you need to
83// annotate your `T` with `alignas` yourself.
84//
85
86// Support thread-local storage, with extra capability to traverse all instances
87// among threads.
88template <class T>
89class ThreadLocal {
90 public:
91 constexpr ThreadLocal() : constructor_([]() { return new T(); }) {}
92
93 template <typename F, std::enable_if_t<std::is_invocable_r_v<T*, F>, int> = 0>
94 explicit ThreadLocal(F&& constructor) : constructor_(std::forward<F>(constructor)) {}
95
96 T* Get() const {
97 // NOT locked, I'm not sure if this is right. (However, it nonetheless
98 // should work without problem and nobody else should make it non-null.)
99 auto&& ptr = tlp_->get();
100 return TRPC_LIKELY(!!ptr) ? ptr : MakeTlp();
101 }
102 T* operator->() const { return Get(); }
103 T& operator*() const { return *Get(); }
104
105 T* Leak() noexcept {
106 std::scoped_lock _(init_lock_);
107 return tlp_->release();
108 }
109
110 void Reset(T* newPtr = nullptr) noexcept {
111 std::scoped_lock _(init_lock_);
112 tlp_->reset(newPtr);
113 }
114
115 // `ForEach` calls `f` with pointer (i.e. `T*`) to each thread-local
116 // instances.
117 //
118 // CAUTION: Called with internal lock held. You may not touch TLS in `f`.
119 template <class F>
120 void ForEach(F&& f) const {
121 std::scoped_lock _(init_lock_);
122 tlp_.ForEach([&](auto* p) {
123 if (*p) {
124 f(p->get());
125 }
126 });
127 }
128
129 private:
130 // non-copyable
131 ThreadLocal(const ThreadLocal&) = delete;
132 ThreadLocal& operator=(const ThreadLocal&) = delete;
133 ThreadLocal(const ThreadLocal&&) = delete;
134 ThreadLocal& operator=(const ThreadLocal&&) = delete;
135
136 [[gnu::noinline]] T* MakeTlp() const {
137 std::scoped_lock _(init_lock_);
138 auto const ptr = constructor_();
139 tlp_->reset(ptr);
140 return ptr;
141 }
142
143 mutable internal::ThreadLocalAlwaysInitialized<std::unique_ptr<T>> tlp_;
144
145 // Synchronizes between `ForEach` and other member methods operating on
146 // `tlp_`.
147 mutable std::mutex init_lock_;
148 Function<T*()> constructor_;
149};

157template <class T>
158ThreadLocalAlwaysInitialized<T>::ThreadLocalAlwaysInitialized()
159 : ThreadLocalAlwaysInitialized([](void* ptr) { new (ptr) T(); }) {}
160
161template <class T>
162template <class F>
163ThreadLocalAlwaysInitialized<T>::ThreadLocalAlwaysInitialized(F&& initializer)
164 : initializer_(std::forward<F>(initializer)) {
165 // Initialize the slot in every thread (that has allocated the slot in its
166 // thread-local object array.).
167 auto slot_initializer = [&](auto index) {
168 // Called with layout lock held. Nobody else should be resizing its own
169 // object array or mutating the (type-specific) global layout.
170
171 // Initialize all slots (if the slot itself has already been allocated in
172 // corresponding thread's object array) immediately so that we don't need to
173 // check for initialization on `Get()`.
174 tls::detail::ObjectArrayRegistry<T>::Instance()->ForEachLocked(
175 index, [&](auto* p) { p->objects.InitializeAt(index, initializer_); });
176 };
177
178 // Allocate a slot.
179 offset_ = tls::detail::ObjectArrayLayout<T>::Instance()->CreateEntry(&initializer_, slot_initializer) * sizeof(T);
180}
181
182template <class T>
183ThreadLocalAlwaysInitialized<T>::~ThreadLocalAlwaysInitialized() {
184 TRPC_ASSERT(offset_ % sizeof(T) == 0);
185 auto index = offset_ / sizeof(T);
186
187 // The slot is freed after we have destroyed all instances.
188 tls::detail::ObjectArrayLayout<T>::Instance()->FreeEntry(index, [&] {
189 // Called with layout lock held.
190
191 // Destroy all instances. (We actually reconstructed a new one at the place
192 // we were at.)
193 tls::detail::ObjectArrayRegistry<T>::Instance()->ForEachLocked(index,
194 [&](auto* p) { p->objects.DestroyAt(index); });
195 });
196}
197
198template <class T>
199inline T* ThreadLocalAlwaysInitialized<T>::Get() const {
200 return tls::detail::GetLocalObjectArrayAt<T>(offset_);
201}
202
203template <class T>
204template <class F>
205void ThreadLocalAlwaysInitialized<T>::ForEach(F&& f) const {
206 TRPC_ASSERT(offset_ % sizeof(T) == 0);
207 auto index = offset_ / sizeof(T);
208
209 tls::detail::ObjectArrayRegistry<T>::Instance()->ForEachLocked(index, [&](auto* p) { f(p->objects.GetAt(index)); });
210}
