31// You need to specialize this traits unless your type is inherited from
32// `RefCounted<T>`.
33template <class T, class = void>
34struct RefTraits {
35 // Increment reference counter on `ptr` with `std::memory_order_relaxed`.
36 static void Reference(T* ptr) noexcept {
37 static_assert(sizeof(T) == 0,
38 "To use `RefPtr<T>`, you need either inherit from `RefCounted<T>` "
39 "(which you didn't), or specialize `RefTraits<T>`.");
40 }
41
42 // Decrement reference counter on `ptr` with `std::memory_order_acq_rel`.
43 //
44 // If the counter after decrement reaches zero, resource allocated for `ptr`
45 // should be released.
46 static void Dereference(T* ptr) noexcept {
47 static_assert(sizeof(T) == 0,
48 "To use `RefPtr<T>`, you need either inherit from `RefCounted<T>` "
49 "(which you didn't), or specialize `RefTraits<T>`.");
50 }
51};

53// For types inherited from `RefCounted`, they're supported by `RefPtr` (or
54// `RefTraits<T>`, to be accurate) by default.
55//
56// Default constructed one has a reference count of one (every object whose
57// reference count reaches zero should have already been destroyed). Should you
58// want to construct a `RefPtr`, use overload that accepts `adopt_ptr`.
59//
60// IMPLEMENTATION DETAIL: For the sake of implementation simplicity, at the
61// moment `Deleter` is declared as a friend to `RefCounted<T>`. This declaration
62// is subject to change.
63template <class T, class Deleter = std::default_delete<T>>
64class RefCounted {
65 public:
66 // Increment ref-count.
67 constexpr void Ref() noexcept;
68
69 // Decrement ref-count, if it reaches zero after the decrement, the pointer is
70 // freed by `Deleter`.
71 constexpr void Deref() noexcept;
72
73 // Get current ref-count.
74 //
75 // It's unsafe, as by the time the ref-count is returned, it may well have
76 // changed. You cannot rely on it for accurate ref-count even if it
77 // returns 1, because the implementation loads with memory_order_relaxed.
78 // This method is similiar to `std::shared_ptr<T>::use_count`.
79 constexpr std::uint32_t UnsafeRefCount() const noexcept;
80
81 protected:
82 // `RefCounted` must be inherited.
83 RefCounted() = default;
84
85 // Destructor is NOT defined as virtual on purpose, we always cast `this` to
86 // `T*` before calling `delete`.
87
88 private:
89 friend Deleter;
90
91 // Hopefully `std::uint32_t` is large enough to store a ref count.
92 //
93 // We tried using `std::uint_fast32_t` here, it uses 8 bytes. I'm not aware of
94 // the benefit of using an 8-byte integer here (at least for the purpose of
95 // reference-couting.)
96 std::atomic<std::uint32_t> ref_count_{1};
97};

99// Keep the overhead the same as an atomic `std::uint3232_t`.
100static_assert(sizeof(RefCounted<int>) == sizeof(std::atomic<std::uint32_t>));

106// This methods serves multiple purposes:
107//
108// - If `T` is explicitly specified, it tests if `T` inherits from
109// `RefCounted<T, ...>`, and returns `RefCounted<T, ...>`;
110//
111// - If `T` is not specified, it tests if `T` inherits from some `RefCounted<U,
112// ...>`, and returns `RefCounted<U, ...>`;
113//
114// - Otherwise `int` is returned, which can never inherits from `RefCounted<int,
115// ...>`.
116template <class T, class... Us>
117RefCounted<T, Us...> GetRefCountedType(const RefCounted<T, Us...>*);
118template <class T>
119int GetRefCountedType(...);
120int GetRefCountedType(...); // Match failure.
121
122// Extract `T` from `RefCounted<T, ...>`.
123//
124// We're returning `std::common_type<T>` instead of `T` so as not to require `T`
125// to be complete.
126template <class T, class... Us>
127std::common_type<T> GetRefeeType(const RefCounted<T, Us...>*);
128std::common_type<int> GetRefeeType(...);
129
130template <class T>
131using refee_t = typename decltype(GetRefeeType(reinterpret_cast<const T*>(0)))::type;
132
133// If `T` inherits from `RefCounted<T, ...>` directly, `T` is returned,
134// otherwise `int` (which can never inherits `RefCounted<int, ...>`) is
135// returned.
136template <class T>
137using direct_refee_t = refee_t<decltype(GetRefCountedType<T>(reinterpret_cast<const T*>(0)))>;
138
139// Test if `T` is a subclass of some `RefCounted<T, ...>`.
140template <class T>
141constexpr auto is_ref_counted_directly_v = !std::is_same_v<int, direct_refee_t<T>>;
142
143// If `T` is inheriting from some `RefCounted<U, ...>` unambiguously, this
144// method returns `U`. Otherwise it returns `int`.
145template <class T>
146using indirect_refee_t = refee_t<decltype(GetRefCountedType(reinterpret_cast<const T*>(0)))>;
147
148// Test if:
149//
150// - `T` is inherited from `RefCounted<U, ...>` indirectly; (1)
151// - `U` is inherited from `RefCounted<U, ...>` directly; (2)
152// - and `U` has a virtual destructor. (3)
153//
154// In this case, we can safely use `RefTraits<RefCounted<U, ...>>` on
155// `RefCounted<T>`.
156template <class T>
157constexpr auto is_ref_counted_indirectly_safe_v = !is_ref_counted_directly_v<T> && // 1
158 is_ref_counted_directly_v<indirect_refee_t<T>> && // 2
159 std::has_virtual_destructor_v<indirect_refee_t<T>>; // 3
160
161// Test if `RefTraits<as_ref_counted_t<T>>` is safe for `T`.
162//
163// We consider default traits as safe if either:
164//
165// - `T` is directly inheriting `RefCounted<T, ...>` (so its destructor does not
166// have to be `virtual`), or
167// - `T` inherts from `RefCounted<U, ...>` and `U`'s destructor is virtual.
168template <class T>
169constexpr auto is_default_ref_traits_safe_v =
170 detail::is_ref_counted_directly_v<T> || detail::is_ref_counted_indirectly_safe_v<T>;
171
172template <class T>
173using as_ref_counted_t = decltype(GetRefCountedType(reinterpret_cast<const T*>(0)));
174
175} // namespace detail

179struct RefTraits<RefCounted<T, Deleter>> {

190// Our own design of `retain_ptr`, with several naming changes to make it easier
191// to understand.
192//
193// @sa: http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2018/p0468r1.html
194template <class T>
195class RefPtr {
196 using Traits = RefTraits<T>;
197
198 public:
199 using element_type = T;
200
201 // Default constructed one does not own a pointer.
202 constexpr RefPtr() noexcept : ptr_(nullptr) {}
203
204 // Same as default-constructed one.
205 /* implicit */ constexpr RefPtr(std::nullptr_t) noexcept : ptr_(nullptr) {}
206
207 // Reference counter is decrement on destruction.
208 ~RefPtr() {
209 if (ptr_) {
210 Traits::Dereference(ptr_);
211 }
212 }
213
214 // Increment reference counter on `ptr` and hold it.
215 //
216 // P0468 declares `xxx_t` as the second argument, but I'd still prefer to
217 // place it as the first one. `std::scoped_lock` revealed the shortcoming of
218 // placing `std::adopt_lock_t` as the last parameter, I won't repeat that
219 // error here.
220 constexpr RefPtr(ref_ptr_t, T* ptr) noexcept;
221
222 // Hold `ptr` without increasing its reference counter.
223 constexpr RefPtr(adopt_ptr_t, T* ptr) noexcept : ptr_(ptr) {}
224
225 // TBH this is a dangerous conversion constructor. Even if `T` does not have
226 // virtual destructor.
227 //
228 // However, testing if `T` has a virtual destructor comes with a price: we'll
229 // require `T` to be complete when defining `RefPtr<T>`, which is rather
230 // annoying.
231 //
232 // Given that `std::unique_ptr` does not test destructor's virtualization
233 // state either, we ignore it for now.
234 template <class U, class = std::enable_if_t<std::is_convertible_v<U*, T*>>>
235 constexpr RefPtr(RefPtr<U> ptr) noexcept : ptr_(ptr.Leak()) {}
236
237 // Copyable, movable.
238 constexpr RefPtr(const RefPtr& ptr) noexcept;
239 constexpr RefPtr(RefPtr&& ptr) noexcept;
240 constexpr RefPtr& operator=(const RefPtr& ptr) noexcept;
241 constexpr RefPtr& operator=(RefPtr&& ptr) noexcept;
242
243 // `boost::intrusive_ptr(T*)` increments reference counter, while
244 // `std::ref_ptr(T*)` (as specified by the proposal) does not. The
245 // inconsistency between them can be confusing. To be perfect clear, we do not
246 // support `RefPtr(T*)`.
247 //
248 // RefPtr(T*) = delete;
249
250 // Accessors.
251 constexpr T* operator->() const noexcept { return Get(); }
252 constexpr T& operator*() const noexcept { return *Get(); }
253 constexpr T* Get() const noexcept { return ptr_; }
254 // Compatible with the usage of std::shared_ptr.
255 constexpr T* get() const noexcept { return Get(); }
256
257 // Test if *this holds a pointer.
258 constexpr explicit operator bool() const noexcept { return ptr_; }
259 bool operator!=(std::nullptr_t) const noexcept { return ptr_ != nullptr; }
260 bool operator==(std::nullptr_t) const noexcept { return ptr_ == nullptr; }
261
262 // Equivalent to `Reset()`.
263 constexpr RefPtr& operator=(std::nullptr_t) noexcept;
264
265 // Reset *this to an empty one.
266 constexpr void Reset() noexcept;
267
268 // Release whatever *this currently holds and hold `ptr` instead.
269 constexpr void Reset(ref_ptr_t, T* ptr) noexcept;
270 constexpr void Reset(adopt_ptr_t, T* ptr) noexcept;
271
272 // Gives up ownership on its internal pointer, which is returned.
273 //
274 // The caller is responsible for calling `RefTraits<T>::Dereference` when it
275 // sees fit.
276 [[nodiscard]] constexpr T* Leak() noexcept { return std::exchange(ptr_, nullptr); }
277
278 private:
279 T* ptr_ = nullptr;
280};

289template <class T, class Deleter>
290constexpr void RefCounted<T, Deleter>::Ref() noexcept {
291 ref_count_.fetch_add(1, std::memory_order_relaxed);
292}

294template <class T, class Deleter>
295constexpr void RefCounted<T, Deleter>::Deref() noexcept {
296 // It seems that we can simply test if `ref_count_` is 1, and save an atomic
297 // operation if it is (as we're the only reference holder). However I don't
298 // see a perf. boost in implementing it, so I keep it unchanged. we might want
299 // to take a deeper look later.
300 if (ref_count_.fetch_sub(1, std::memory_order_acq_rel) == 1) {
301 Deleter()(static_cast<T*>(this));
302 }
303}

329template <class T>
330constexpr RefPtr<T>& RefPtr<T>::operator=(const RefPtr& ptr) noexcept {
331 if (TRPC_UNLIKELY(&ptr == this)) {
332 return *this;
333 }
334 if (ptr_) {
335 Traits::Dereference(ptr_);
336 }
337 ptr_ = ptr.ptr_;
338 if (ptr_) {
339 Traits::Reference(ptr_);
340 }
341 return *this;
342}

395}

397template <typename _Tp, typename _Up>
398inline RefPtr<_Tp> dynamic_pointer_cast(const RefPtr<_Up>& __r) noexcept {
399 using _Sp = RefPtr<_Tp>;
400 if (auto* __p = dynamic_cast<typename _Sp::element_type*>(__r.get())) {
401 return _Sp(ref_ptr, __p);
402 }
403 return _Sp();
