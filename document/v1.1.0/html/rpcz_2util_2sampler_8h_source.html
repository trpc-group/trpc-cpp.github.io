89 bool Sample(uint32_t id) {
90 // Hitting sample rate.
91 if (sampling_rate_ > 0 && id % sampling_rate_ == 0) {
92 // But current window bigger than high water, stop sample.
93 if (sample_rate_limiter_.Increase() > speed_max_rate_) {
94 return false;
95 } else {
96 // Otherwise, do sample.
97 return true;
98 }
99 } else {
100 // Not hitting sample rate, but current window count is less than low water, sample anyway.
101 if (sample_rate_limiter_.GetSampleNum() < speed_min_rate_) {
102 sample_rate_limiter_.Increase();
103 return true;
104 } else {
105 // Otherwise, stop sample.
106 return false;
107 }
108 }
109 }
110
111 private:
112 // Sample rate, may be 0.
113 uint32_t sampling_rate_;
114 // Low water.
115 uint32_t speed_min_rate_;
116 // High water.
117 uint32_t speed_max_rate_;
118 // Limiter by window.
119 SampleRateLimiter sample_rate_limiter_;
120};
