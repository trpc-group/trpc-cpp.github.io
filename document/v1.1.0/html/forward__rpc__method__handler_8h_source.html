28 public:

33
34 void Execute(const ServerContextPtr& context, NoncontiguousBuffer&& req_body,
35 NoncontiguousBuffer& rsp_body) noexcept override {
36 uint32_t encode_type = context->GetReqEncodeType();
37
38 TRPC_LOG_TRACE("encode_type:" << encode_type);
39
40 Status status = forward_func_(context, std::move(req_body), rsp_body);
41
42 if (context->IsResponse()) {
43 if (!status.OK()) {
44 context->SetStatus(std::move(status));
45 }
46 }
47 }
48
49 void Execute(const ServerContextPtr& context) noexcept override { TRPC_ASSERT(false && "Unreachable"); }

51 private:
