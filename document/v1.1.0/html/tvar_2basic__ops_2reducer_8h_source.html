59template <typename T, typename Base = WriteMostly<AddTraits<T>>>
60class Counter : public Base {
61 public:
63 explicit Counter(add_cr_non_integral_t<T> init_value = T()) : Base(init_value, init_value) {}
64
66 explicit Counter(std::string_view rel_path, add_cr_non_integral_t<T> init_value = T())
67 : Counter(init_value) {
68 handle_ = Base::LinkToParent(rel_path);
69 }
70
72 Counter(TrpcVarGroup* parent, std::string_view rel_path,
73 add_cr_non_integral_t<T> init_value = T())
74 : Counter(init_value) {
75 handle_ = Base::LinkToParent(rel_path, parent);
76 }
77
79 void Add(add_cr_non_integral_t<T> value) noexcept {
80 TRPC_ASSERT(value >= 0);
81 Base::Update(value);
82 }
83
85 void Increment() noexcept { Add(1); }
86
88 std::string GetAbsPath() const { return IsExposed() ? handle_->abs_path : std::string(); }
