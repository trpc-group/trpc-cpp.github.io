46inline uint64_t RoundOfExpectation(uint64_t a, uint64_t b) {
47 if (TRPC_UNLIKELY(b == 0)) {
48 return 0;
49 }
50 return a / b + (FastRandLessThan(b) < a % b);
51}

55template <size_t SAMPLE_SIZE>
56class PercentileInterval {
57 public:
58 PercentileInterval() = default;
59
61 uint32_t GetSampleAt(size_t index) {
62 const size_t saved_num = num_samples_;
63 if (index >= saved_num) {
64 if (saved_num == 0) {
65 return 0;
66 }
67 index = saved_num - 1;
68 }
69 if (!sorted_) {
70 std::sort(samples_, samples_ + saved_num);
71 sorted_ = true;
72 }
73 TRPC_ASSERT(saved_num == num_samples_);
74 return samples_[index];
75 }
76
78 template <size_t size2>
79 void Merge(PercentileInterval<size2> const& rhs) {
80 if (rhs.num_added_ == 0) {
81 return;
82 }
83 static_assert(SAMPLE_SIZE >= size2, "must merge small interval into larger one currently");
84 TRPC_ASSERT(rhs.num_samples_ == rhs.num_added_);
85 // Fully merged.
86 if (num_added_ + rhs.num_added_ <= SAMPLE_SIZE) {
87 TRPC_ASSERT(num_samples_ == num_added_);
88 memcpy(samples_ + num_samples_, rhs.samples_, sizeof(samples_[0]) * rhs.num_samples_);
89 num_samples_ += rhs.num_samples_;
90 } else {
91 size_t num_remain = RoundOfExpectation(num_added_ * SAMPLE_SIZE, num_added_ + rhs.num_added_);
92 TRPC_ASSERT(num_remain <= num_samples_);
93 for (size_t i = num_samples_; i > num_remain; --i) {
94 samples_[FastRandLessThan(i)] = samples_[i - 1];
95 }
96 const size_t num_remain_from_rhs = SAMPLE_SIZE - num_remain;
97 TRPC_ASSERT(num_remain_from_rhs <= rhs.num_samples_);
98 TRPC_DEFINE_SMALL_ARRAY(uint32_t, tmp, rhs.num_samples_, 64);
99 memcpy(tmp, rhs.samples_, sizeof(uint32_t) * rhs.num_samples_);
100 for (size_t i = 0; i < num_remain_from_rhs; ++i) {
101 const int index = FastRandLessThan(rhs.num_samples_ - i);
102 samples_[num_remain++] = tmp[index];
103 tmp[index] = tmp[rhs.num_samples_ - i - 1];
104 }
105 num_samples_ = num_remain;
106 TRPC_ASSERT(num_samples_ == SAMPLE_SIZE);
107 }
108 num_added_ += rhs.num_added_;
109 }
110
112 template <size_t size2>
113 void MergeWithExpectation(PercentileInterval<size2>& mutable_rhs, size_t n) {
114 TRPC_ASSERT(n <= mutable_rhs.num_samples_);
115 num_added_ += mutable_rhs.num_added_;
116 // Fully merged.
117 if (num_samples_ + n <= SAMPLE_SIZE && n == mutable_rhs.num_samples_) {
118 memcpy(samples_ + num_samples_, mutable_rhs.samples_, sizeof(samples_[0]) * n);
119 num_samples_ += n;
120 return;
121 }
122 // Randomly selected.
123 for (size_t i = 0; i < n; ++i) {
124 size_t index = FastRandLessThan(mutable_rhs.num_samples_ - i);
125 if (num_samples_ < SAMPLE_SIZE) {
126 samples_[num_samples_++] = mutable_rhs.samples_[index];
127 } else {
128 samples_[FastRandLessThan(num_samples_)] = mutable_rhs.samples_[index];
129 }
130 std::swap(mutable_rhs.samples_[index],
131 mutable_rhs.samples_[mutable_rhs.num_samples_ - i - 1]);
132 }
133 }

137 bool Add32(uint32_t x) {
138 if (TRPC_UNLIKELY(num_samples_ >= SAMPLE_SIZE)) {
139 TRPC_FMT_ERROR("This interval was full");
140 return false;
141 }
142 ++num_added_;
143 samples_[num_samples_++] = x;
144 return true;
145 }
146
148 void Clear() {
149 num_added_ = 0;
150 sorted_ = false;
151 num_samples_ = 0;
152 }
153
155 [[nodiscard]] bool Full() const { return num_samples_ == SAMPLE_SIZE; }
156
158 [[nodiscard]] bool Empty() const { return !num_samples_; }
159
161 [[nodiscard]] uint32_t AddedCount() const { return num_added_; }
162
164 [[nodiscard]] uint32_t SampleCount() const { return num_samples_; }
165
168 bool operator==(const PercentileInterval& rhs) const {
169 return (num_added_ == rhs.num_added_ && num_samples_ == rhs.num_samples_ &&
170 memcmp(samples_, rhs.samples_, num_samples_ * sizeof(uint32_t)) == 0);
171 }
172
173 private:
174 template <size_t size2>
175 friend class PercentileInterval;
176
177 static_assert(SAMPLE_SIZE <= 65536, "SAMPLE_SIZE must be 16bit");
178 // How many times add.
179 uint32_t num_added_{0};
180 // Is samples sorted.
181 bool sorted_{false};
182 // How many time sampled.
183 uint16_t num_samples_{0};
184 // Container storing samples.
185 uint32_t samples_[SAMPLE_SIZE]{0};
186};

194template <size_t SAMPLE_SIZE_IN>

213 PercentileSamples(const PercentileSamples& rhs) {
214 num_added_ = rhs.num_added_;
215 for (size_t i = 0; i < NUM_INTERVALS; ++i) {
216 if (rhs.intervals_[i] && !rhs.intervals_[i]->Empty()) {
217 intervals_[i] = new PercentileInterval<SAMPLE_SIZE>(*rhs.intervals_[i]);
218 } else {
219 intervals_[i] = nullptr;
220 }
221 }
222 }

226 void operator=(const PercentileSamples& rhs) {
227 num_added_ = rhs.num_added_;
228 for (size_t i = 0; i < NUM_INTERVALS; ++i) {
229 if (rhs.intervals_[i] && !rhs.intervals_[i]->Empty()) {
230 GetIntervalAt(i) = *rhs.intervals_[i];
231 } else if (intervals_[i]) {
232 intervals_[i]->Clear();

246 uint32_t GetNumber(double ratio) {
247 auto n = static_cast<size_t>(ceil(ratio * num_added_));
248 if (n > num_added_) {
249 n = num_added_;
250 } else if (n == 0) {
251 return 0;
252 }
253 for (auto&& interval : intervals_) {
254 if (interval == nullptr) {
255 continue;
256 }
257 PercentileInterval<SAMPLE_SIZE>& invl = *interval;
258 if (n <= invl.AddedCount()) {
259 size_t sample_n = n * invl.SampleCount() / invl.AddedCount();
260 size_t sample_index = (sample_n ? sample_n - 1 : 0);
261 return invl.GetSampleAt(sample_index);
262 }
263 n -= invl.AddedCount();
264 }
265 TRPC_ASSERT(false && "Can't reach here");
266 return std::numeric_limits<uint32_t>::max();
267 }

270 template <size_t size2>
271 void Merge(const PercentileSamples<size2>& rhs) {
272 num_added_ += rhs.num_added_;
273 for (size_t i = 0; i < NUM_INTERVALS; ++i) {
274 if (rhs.intervals_[i] && !rhs.intervals_[i]->Empty()) {
275 GetIntervalAt(i).Merge(*rhs.intervals_[i]);
276 }
277 }
278 }

281 template <typename Iterator>
282 void CombineOf(const Iterator& begin, const Iterator& end) {
283 if (num_added_) {
284 // Very unlikely.
285 for (auto&& interval : intervals_) {
286 if (interval) {
287 interval->Clear();
288 }
289 }
290 num_added_ = 0;
291 }
292
293 for (Iterator iter = begin; iter != end; ++iter) {
294 num_added_ += iter->num_added_;
295 }
296
297 // Calculate probabilities for each interval.
298 for (size_t i = 0; i < NUM_INTERVALS; ++i) {
299 size_t total = 0;
300 size_t total_sample = 0;
301 for (Iterator iter = begin; iter != end; ++iter) {
302 if (iter->intervals_[i]) {
303 total += iter->intervals_[i]->AddedCount();
304 total_sample += iter->intervals_[i]->SampleCount();
305 }
306 }
307 if (total == 0) {
308 // Empty interval.
309 continue;
310 }
311
312 // Consider that sub interval took |a| samples out of |b| totally,
313 // each sample won the probability of |a/b| according to the
314 // algorithm of add32(), now we should pick some samples into the
315 // combined interval that satisfied each sample has the
316 // probability of |SAMPLE_SIZE/total|, so each sample has the
317 // probability of |(SAMPLE_SIZE*b)/(a*total) to remain and the
318 // expected number of samples in this interval is
319 // |(SAMPLE_SIZE*b)/total|
320 for (Iterator iter = begin; iter != end; ++iter) {
321 if (!iter->intervals_[i] || iter->intervals_[i]->Empty()) {
322 continue;
323 }
324 typename std::add_lvalue_reference_t<decltype(*(iter->intervals_[i]))> invl =
325 *(iter->intervals_[i]);
326 if (total <= SAMPLE_SIZE) {
327 GetIntervalAt(i).MergeWithExpectation(invl, invl.SampleCount());
328 continue;
329 }
330 // Each
331 const size_t b = invl.AddedCount();
332 const size_t remain =
333 std::min(RoundOfExpectation(b * SAMPLE_SIZE, total), (size_t)invl.SampleCount());
334 GetIntervalAt(i).MergeWithExpectation(invl, remain);
335 }
336 }
337 }

340 bool operator==(const PercentileSamples& rhs) const {
341 for (size_t i = 0; i < NUM_INTERVALS; ++i) {
342 if (intervals_ != rhs.intervals_[i]) {
343 return false;
344 }
345 }
346 return true;
347 }

359 intervals_[index] = new PercentileInterval<SAMPLE_SIZE>;

365 size_t num_added_{0};
366 PercentileInterval<SAMPLE_SIZE>* intervals_[NUM_INTERVALS]{nullptr};

382class PercentileTraits {
383 public:
384 struct AddPercentileSamples {
385 template <size_t size1, size_t size2>
386 void operator()(PercentileSamples<size1>* b1, PercentileSamples<size2> const& b2) const {
387 b1->Merge(b2);
388 }
389 };
390
391 public:
392 using Type = ThreadLocalPercentileSamples;
393 using WriteBuffer = AtomicType<Type>;
394 using InputDataType = std::uint32_t;
395 using ResultType = GlobalPercentileSamples;
396 using SamplerOp = AddPercentileSamples;
397 using SamplerInvOp = VoidOp;
398
399 public:
400 PercentileTraits(PercentileTraits const&) = delete;
401
402 void operator=(PercentileTraits const&) = delete;
403
404 PercentileTraits() = default;

407 template <typename TLS>
408 static void Update(TLS* wb, InputDataType val) {
409 GlobalValue<TLS> g(wb);

414 static void Merge(ResultType* wb1, Type const& wb2) { AddPercentileSamples()(wb1, wb2); }
