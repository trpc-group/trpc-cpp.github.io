99 // Select the default EncodeType and EncodeDataType base on the message type.
100 // Note: The default matching can only be done for general types, and if it is a unknown user-defined type, the
101 // EncodeType needs to be set explicitly.
102 template <class Message>
103 bool SelectDefalutEncodeType(serialization::SerializationType& encode_type, serialization::DataType& data_type) {
104 if constexpr (std::is_convertible_v<Message*, google::protobuf::MessageLite*>) {
105 data_type = serialization::kPbMessage;
106 encode_type = serialization::kPbType;
107 } else if constexpr (std::is_convertible_v<Message*, flatbuffers::trpc::MessageFbs*>) {
108 data_type = serialization::kFlatBuffers;
109 encode_type = serialization::kFlatBuffersType;
110 } else if constexpr (std::is_convertible_v<Message*, rapidjson::Document*>) {
111 data_type = serialization::kRapidJson;
112 encode_type = serialization::kJsonType;
113 } else if constexpr (std::is_convertible_v<Message*, std::string*>) {
114 data_type = serialization::kStringNoop;
115 encode_type = serialization::kNoopType;
116 } else if constexpr (std::is_convertible_v<Message*, NoncontiguousBuffer*>) {
117 data_type = serialization::kNonContiguousBufferNoop;
118 encode_type = serialization::kNoopType;
119 } else {
120 return false;
121 }
122
123 return true;
124 }

126 template <class RequestMessage>
127 bool SetReqEncode(const ClientContextPtr& context) {
128 if (TRPC_UNLIKELY(context->IsTransparent())) {
129 return true;
130 }
131
132 serialization::DataType selected_data_type;
133 serialization::SerializationType selected_encode_type;
134 if (!SelectDefalutEncodeType<RequestMessage>(selected_encode_type, selected_data_type)) {
135 return false;
136 }
137
138 context->SetReqEncodeDataType(selected_data_type);
139 // If the serialization type is not set in the context (default is PB type), the specific serialization type is
140 // inferred based on the type of the request body.
141 if (context->GetReqEncodeType() == serialization::kPbType) {
142 context->SetReqEncodeType(selected_encode_type);
143 }
144
145 return true;
146 }

150 if (TRPC_UNLIKELY(context->IsTransparent())) {
151 return true;

162 context->SetRspEncodeType(selected_encode_type);

183template <class RequestMessage, class ResponseMessage>
184Status RpcServiceProxy::UnaryInvoke(const ClientContextPtr& context, const RequestMessage& req, ResponseMessage* rsp) {
185 TRPC_ASSERT(context->GetRequest() != nullptr);
186
187 // Set the corresponding encoding based on the types of the request and response.
188 if (TRPC_UNLIKELY(!(SetMessageEncodeType<RequestMessage, ResponseMessage>)(context))) {
189 return context->GetStatus();
190 }
191
192 FillClientContext(context);
193
194 // Set user request data struct
195 context->SetRequestData(const_cast<RequestMessage*>(&req));
196 // Set user response data struct
197 context->SetResponseData(rsp);
198
199 // Execute pre-RPC invoke filtes
200 int filter_ret = RunFilters(FilterPoint::CLIENT_PRE_RPC_INVOKE, context);
201 if (filter_ret == 0) {
202 UnaryInvokeImp<RequestMessage, ResponseMessage>(context, req, rsp);
203 }
204 // Execute post-RPC invoke filtes
205 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
206
207 context->SetRequestData(nullptr);
208 context->SetResponseData(nullptr);
209
210 context->SetEndTimestampUs(trpc::time::GetMicroSeconds());
211
212 return context->GetStatus();
213}

215template <class RequestMessage, class ResponseMessage>
216void RpcServiceProxy::UnaryInvokeImp(const ClientContextPtr& context, const RequestMessage& req, ResponseMessage* rsp) {
217 const ProtocolPtr& req_protocol = context->GetRequest();
218 if (!codec_->FillRequest(context, req_protocol, reinterpret_cast<void*>(const_cast<RequestMessage*>(&req)))) {
219 std::string error("service name:");
220 error += GetServiceName();
221 error += ", request fill body failed.";
222 TRPC_LOG_ERROR(error);
223
224 Status status;
225 status.SetFrameworkRetCode(TrpcRetCode::TRPC_CLIENT_ENCODE_ERR);
226 status.SetErrorMessage(error);
227
228 context->SetStatus(std::move(status));
229
230 return;
231 }
232
233 ProtocolPtr& rsp_protocol = context->GetResponse();
234 Status unary_invoke_status = ServiceProxy::UnaryInvoke(context, req_protocol, rsp_protocol);
235
236 if (TRPC_UNLIKELY(!unary_invoke_status.OK())) {
237 return;
238 }
239
240 if (!codec_->FillResponse(context, rsp_protocol, static_cast<void*>(rsp))) {
241 std::string error("service name:");
242 error += GetServiceName();
243 error += ", response fill body failed.";
244 TRPC_LOG_TRACE(error);
245
246 if (context->GetStatus().OK()) {
247 Status status;
248 status.SetFrameworkRetCode(TrpcRetCode::TRPC_CLIENT_DECODE_ERR);
249 status.SetErrorMessage(error);
250 context->SetStatus(std::move(status));
251 }
252 }

255template <class RequestMessage, class ResponseMessage>
256Future<ResponseMessage> RpcServiceProxy::AsyncUnaryInvoke(const ClientContextPtr& context, const RequestMessage& req) {
257 TRPC_ASSERT(context->GetRequest() != nullptr);
258
259 if (TRPC_UNLIKELY(!(SetMessageEncodeType<RequestMessage, ResponseMessage>)(context))) {
260 return MakeExceptionFuture<ResponseMessage>(CommonException(
261 context->GetStatus().ErrorMessage().c_str(), static_cast<int>(context->GetStatus().GetFrameworkRetCode())));
262 }
263
264 FillClientContext(context);
265
266 context->SetRequestData(const_cast<RequestMessage*>(&req));
267
268 int filter_ret = RunFilters(FilterPoint::CLIENT_PRE_RPC_INVOKE, context);
269 if (filter_ret == 0) {
270 context->SetRequestData(nullptr);
271 // Logic for execute post-RPC invoke filters is processed in AsyncUnaryTransportInvoke.
272 // To reduce the number of layers of Future Then calls for performance optimization .
273 return AsyncUnaryInvokeImp<RequestMessage, ResponseMessage>(context, req);
274 }
275
276 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
277
278 context->SetRequestData(nullptr);
279 context->SetEndTimestampUs(trpc::time::GetMicroSeconds());
280
281 const Status& result = context->GetStatus();
282
283 return MakeExceptionFuture<ResponseMessage>(CommonException(result.ErrorMessage().c_str()));
284}

286template <class RequestMessage, class ResponseMessage>
287Future<ResponseMessage> RpcServiceProxy::AsyncUnaryInvokeImp(const ClientContextPtr& context,
288 const RequestMessage& req) {
289 ProtocolPtr& req_protocol = context->GetRequest();
290 if (!codec_->FillRequest(context, req_protocol, reinterpret_cast<void*>(const_cast<RequestMessage*>(&req)))) {
291 std::string error("service name:");
292 error += GetServiceName();
293 error += ", request fill body failed.";
294
295 TRPC_LOG_ERROR(error);
296
297 Status status;
298 status.SetFrameworkRetCode(TrpcRetCode::TRPC_CLIENT_ENCODE_ERR);
299 status.SetErrorMessage(error);
300
301 context->SetStatus(std::move(status));
302
303 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
304
305 context->SetEndTimestampUs(trpc::time::GetMicroSeconds());
306
307 return MakeExceptionFuture<ResponseMessage>(
308 CommonException(context->GetStatus().ErrorMessage().c_str(), TrpcRetCode::TRPC_CLIENT_ENCODE_ERR));
309 }
310
311 return ServiceProxy::AsyncUnaryInvoke(context, req_protocol)
312 .Then([this, context](Future<ProtocolPtr>&& rsp_protocol) {
313 if (rsp_protocol.IsFailed()) {
314 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
315
316 return MakeExceptionFuture<ResponseMessage>(rsp_protocol.GetException());
317 }
318
319 context->SetResponse(rsp_protocol.GetValue0());
320
321 ResponseMessage rsp_obj;
322 void* raw_rsp = static_cast<void*>(&rsp_obj);
323
324 if (!codec_->FillResponse(context, context->GetResponse(), raw_rsp)) {
325 std::string error("service name:");
326 error += GetServiceName();
327 error += ", response fill body failed.";
328
329 TRPC_LOG_TRACE(error);
330
331 if (context->GetStatus().OK()) {
332 Status status;
333 status.SetFrameworkRetCode(TrpcRetCode::TRPC_CLIENT_DECODE_ERR);
334 status.SetErrorMessage(error);
335 context->SetStatus(std::move(status));
336 }
337 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
338
339 context->SetEndTimestampUs(trpc::time::GetMicroSeconds());
340
341 return MakeExceptionFuture<ResponseMessage>(UnaryRpcError(context->GetStatus()));
342 }
343
344 // Set the response data (for use by the CLIENT_POST_RPC_INVOKE filter for instrumentation) when the call is
345 // successful.
346 context->SetResponseData(&rsp_obj);
347
348 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
349
350 context->SetResponseData(nullptr);
351 context->SetEndTimestampUs(trpc::time::GetMicroSeconds());
352
353 return MakeReadyFuture<ResponseMessage>(std::move(rsp_obj));
354 });
355}

357template <class RequestProtocol>
358Status RpcServiceProxy::OnewayInvoke(const ClientContextPtr& context, const RequestProtocol& req) {
359 TRPC_ASSERT(context->GetRequest() != nullptr);
360
361 context->SetCallType(kOnewayCall);
362
363 if (TRPC_LIKELY(!SetReqEncode<RequestProtocol>(context))) {
364 context->SetEndTimestampUs(trpc::time::GetMicroSeconds());
365
366 std::string error("not support RequestMessage type.");
367 Status status;
368 status.SetFrameworkRetCode(static_cast<int>(TrpcRetCode::TRPC_CLIENT_ENCODE_ERR));
369 status.SetErrorMessage(std::move(error));
370
371 return status;
372 }
373
374 FillClientContext(context);
375
376 context->SetRequestData(const_cast<RequestProtocol*>(&req));
377
378 auto filter_ret = RunFilters(FilterPoint::CLIENT_PRE_RPC_INVOKE, context);
379 if (filter_ret == 0) {
380 const ProtocolPtr& req_protocol = context->GetRequest();
381
382 if (!codec_->FillRequest(context, req_protocol, reinterpret_cast<void*>(const_cast<RequestProtocol*>(&req)))) {
383 context->SetEndTimestampUs(trpc::time::GetMicroSeconds());
384
385 std::string error("service name:");
386 error += GetServiceName();
387 error += ", request fill body failed.";
388 TRPC_LOG_ERROR(error);
389
390 Status status;
391 status.SetFrameworkRetCode(TrpcRetCode::TRPC_CLIENT_ENCODE_ERR);
392 status.SetErrorMessage(error);
393 context->SetStatus(status);
394
395 return status;
396 }
397
398 ServiceProxy::OnewayInvoke(context, req_protocol);
399 }
400
401 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
402
403 context->SetRequestData(nullptr);
404 context->SetEndTimestampUs(trpc::time::GetMicroSeconds());
405
406 return context->GetStatus();
407}

443template <class W, class R>
444stream::StreamReaderWriter<W, R> RpcServiceProxy::CreateStreamReaderWriter(const ClientContextPtr& context,
445 void* rpc_reply_msg) {
446 auto stream_provider = SelectStreamProvider(context, rpc_reply_msg);
447 if (auto options = stream_provider->GetMutableStreamOptions(); options != nullptr) {
448 options->stream_max_window_size = GetServiceProxyOption()->stream_max_window_size;
449 }
450
451 if (!SetReqEncode<W>(context)) {
452 context->SetReqEncodeDataType(serialization::kMaxDataType);
453 TRPC_FMT_ERROR("Not supported serialization type for request message");
454 }
455 if (!SetRspEncode<R>(context)) {
456 context->SetRspEncodeDataType(serialization::kMaxDataType);
457 TRPC_FMT_ERROR("Not supported serialization type for response message");
458 }
459
460 // Initialize the options for the message content encoder/decoder.
461 stream::MessageContentCodecOptions content_codec_options{
462 .serialization = serialization::SerializationFactory::GetInstance()->Get(context->GetReqEncodeType()),
463 .content_type = context->GetReqEncodeType(),
464 };
465
466 stream_provider->Start();
467
468 return stream::Create<W, R>(stream_provider, content_codec_options);
469}

471template <class W, class R>
472Future<std::pair<stream::AsyncWriterPtr<W>, Future<R>>> RpcServiceProxy::AsyncStreamInvoke(
473 const ClientContextPtr& context) {
474 std::chrono::milliseconds timeout(context->GetTimeout());
475 return CreateAsyncStream<W, R>(context).Then([timeout](stream::AsyncReaderWriterPtr<R, W>&& rw) {
476 return MakeReadyFuture<std::pair<stream::AsyncWriterPtr<W>, Future<R>>>(
477 std::make_pair(rw->Writer(), rw->Read(timeout).Then([](std::optional<R>&& reply) {
478 if (reply)
479 return MakeReadyFuture<R>(std::move(reply.value()));
480 else
481 return MakeExceptionFuture<R>(stream::StreamError(Status(-1, -1, "unexcepted stream eof")));
482 })));
483 });
484}

486template <class W, class R>
487Future<stream::AsyncReaderPtr<R>> RpcServiceProxy::AsyncStreamInvoke(const ClientContextPtr& context, W&& request) {
488 return CreateAsyncStream<W, R>(context).Then([request = std::move(request)](stream::AsyncReaderWriterPtr<R, W>&& rw) {
489 return rw->Write(std::move(request)).Then([rw]() {
490 return MakeReadyFuture<stream::AsyncReaderPtr<R>>(rw->Reader());
491 });

495template <class W, class R>
496Future<stream::AsyncReaderWriterPtr<R, W>> RpcServiceProxy::CreateAsyncStream(const ClientContextPtr& context) {
497 auto stream = SelectStreamProvider(context);
498
499 return stream->AsyncStart().Then([stream, context]() {
500 stream::MessageContentCodecOptions content_codec_options{
501 .serialization = serialization::SerializationFactory::GetInstance()->Get(context->GetReqEncodeType()),
502 .content_type = context->GetReqEncodeType(),
503 };
504
505 auto rw = MakeRefCounted<stream::AsyncReaderWriter<R, W>>(stream, content_codec_options);
506 return MakeReadyFuture<stream::AsyncReaderWriterPtr<R, W>>(std::move(rw));
507 });
