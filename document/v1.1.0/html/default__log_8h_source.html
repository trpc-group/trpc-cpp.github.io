106 template <typename Sink, typename SinkConfig>
107 bool InitSink(const char* logger_name, const char* sink_name) {
108 auto& instance = instances_[logger_name];
109 auto& conf = instance.config;
110 // Get local_file configuration
111 SinkConfig sink_config;
112 // configuration does not exist and returns true
113 if (!YAML::GetDefaultLoggerSinkConfig<SinkConfig>(conf.name, "sinks", sink_name, sink_config)) {
114 return true;
115 }
116 auto sink = MakeRefCounted<Sink>();
117 sink->Init(sink_config);
118 // Add the new sink to the logger's sinks
119 instance.logger->sinks().push_back(sink->SpdSink());
120
121 // The tRPC-Cpp framework logging instance has been configured
122 if (!strcmp(logger_name, kTrpcLogCacheStringDefault)) {
123 initted_trpc_logger_instance_ = true;
124 trpc_logger_instance_ = instance;
125 }
126 return true;
127 }
