45enum class SeriesFrequency {
46 // Smooth in last window.
47 SERIES_IN_WINDOW = 0,
48 // Smooth in last second.
49 SERIES_IN_SECOND = 1
50};
51
54template <typename R, SeriesFrequency series_freq>
55class WindowBase {
56 public:
57 using ReducerSamplerType = typename R::ReducerSamplerType;
58 using OpType = typename ReducerSamplerType::OpType;
59 using DataType = typename ReducerSamplerType::DataType;
60
61 class SeriesSamplerType : public Sampler {
62 public:
63 explicit SeriesSamplerType(WindowBase* owner) : owner_(owner), series_() {}
64
65 ~SeriesSamplerType() override = default;
66
67 void TakeSample() override {
68 if (series_freq == SeriesFrequency::SERIES_IN_SECOND) {
69 // Get one-second window value for PerSecond<>, otherwise the
70 // smoother plot may hide peaks.
71 series_.Append(owner_->GetValue(1));
72 } else {
73 // Get the value inside the full window. get_value(1) is
74 // incorrect when users intend to see aggregated values of
75 // the full window in the plot.
76 series_.Append(owner_->GetValue());
77 }
78 }
79
80 std::unordered_map<std::string, std::vector<DataType>> GetSeries() const noexcept {
81 return series_.GetSeries();
82 }
83
84 private:
85 WindowBase* owner_;
86 Series<DataType, OpType> series_;
87 };
88
89 explicit WindowBase(R* var, time_t window_size = -1)
90 : var_(var),
91 window_size_(
92 window_size > 0
93 ? window_size
94 : trpc::TrpcConfig::GetInstance()->GetGlobalConfig().tvar_config.window_size),
95 reducer_sampler_(var->GetReducerSampler()) {
96 TRPC_ASSERT(reducer_sampler_.lock()->SetWindowSize(window_size_) == 0);
97 }
98
100 virtual ~WindowBase() {
101 if (series_sampler_) {
102 series_sampler_->Destroy();
103 }
104 }
105
106 bool GetSpan(time_t window_size, Sample<DataType>* result) const {
107 if (auto ptr = reducer_sampler_.lock()) {
108 return ptr->GetValue(window_size, result);
109 }
110 return false;
111 }
112
113 bool GetSpan(Sample<DataType>* result) const { return GetSpan(window_size_, result); }
114
115 virtual DataType GetValue(time_t window_size) const noexcept {
116 Sample<DataType> tmp;
117 if (GetSpan(window_size, &tmp)) {
118 return tmp.data;
119 }
120 return DataType();
121 }
122
123 virtual DataType GetValue() const noexcept { return GetValue(window_size_); }
124
125 [[nodiscard]] time_t WindowSize() const { return window_size_; }
126
127 void GetSamples(std::vector<DataType>* samples) const {
128 if (auto ptr = reducer_sampler_.lock()) {
129 samples->clear();
130 samples->reserve(window_size_);
131 return ptr->GetSamples(samples, window_size_);
132 }
133 }
134
135 std::weak_ptr<SeriesSamplerType> GetSeries() { return series_sampler_; }
136
137 std::unordered_map<std::string, std::vector<DataType>> GetSeriesValue() const noexcept {
138 if (series_sampler_) {
139 return series_sampler_->GetSeries();
140 }
141 return std::unordered_map<std::string, std::vector<DataType>>();
142 }
143
144 protected:
145 std::optional<TrpcVarGroup::Handle> LinkToParent(std::string_view rel_path,
146 TrpcVarGroup* parent) {
147 if (CheckSeries()) {
148 return TrpcVarGroup::LinkToParent(
149 rel_path, parent, [this] { return detail::ToJsonValue(GetValue()); },
150 [this] { return detail::ToJsonValue(GetSeriesValue()); });
151 } else {
152 return TrpcVarGroup::LinkToParent(rel_path, parent,
153 [this] { return detail::ToJsonValue(GetValue()); });
154 }
155 }
156
157 std::optional<TrpcVarGroup::Handle> LinkToParent(std::string_view rel_path) {
158 return LinkToParent(rel_path, TrpcVarGroup::FindOrCreate("/"));
159 }
160
161private:
163 bool CheckSeries() {
164 if (!trpc::TrpcConfig::GetInstance()->GetGlobalConfig().tvar_config.save_series) {
165 return false;
166 }
167
168 if (!series_sampler_) {
169 series_sampler_ = std::make_shared<SeriesSamplerType>(this);
170 series_sampler_->Schedule();
171 }
172 return true;
173 }
174
175 protected:
176 R* var_;
177 time_t window_size_;
178 // Must be last.
179 std::weak_ptr<ReducerSamplerType> reducer_sampler_;
180 std::shared_ptr<SeriesSamplerType> series_sampler_;
181};

210template <typename R, SeriesFrequency series_freq = SeriesFrequency::SERIES_IN_SECOND>
211class PerSecond : public WindowBase<R, series_freq> {
212 private:
213 using Base = WindowBase<R, series_freq>;
214 using ReducerSamplerType = typename R::ReducerSamplerType;
215 using DataType = typename ReducerSamplerType::DataType;
216
217 public:
218 explicit PerSecond(R* var, time_t window_size = -1) : Base(var, window_size) {}
219
220 PerSecond(std::string_view rel_path, R* var, time_t window_size = -1)
221 : PerSecond(var, window_size) {
222 handle_ = Base::LinkToParent(rel_path);
223 }
224
225 PerSecond(TrpcVarGroup* parent, std::string_view rel_path, R* var, time_t window_size = -1)
226 : PerSecond(var, window_size) {
227 handle_ = Base::LinkToParent(rel_path, parent);
228 }
229
230 DataType GetValue(time_t window_size) const noexcept override {
231 Sample<DataType> s;
232 if (auto ptr = this->reducer_sampler_.lock()) {
233 ptr->GetValue(window_size, &s);
234 }
235 if (s.time_us <= 0) {
236 return static_cast<DataType>(0);
237 }
238 if (std::is_floating_point_v<DataType>) {
239 return static_cast<DataType>(s.data * 1000000.0 / s.time_us);
240 } else {
241 return static_cast<DataType>(round(s.data * 1000000.0 / s.time_us));
242 }
243 }
244
245 DataType GetValue() const noexcept override { return Base::GetValue(); }
246
247 std::string GetAbsPath() const { return IsExposed() ? handle_->abs_path : std::string(); }
248
249 bool IsExposed() const { return handle_.has_value(); }
250
251 private:
252 std::optional<TrpcVarGroup::Handle> handle_;
253};
