34ServerContextPtr MakeServerContext(Service* service, ServerCodec* codec) {
35 TRPC_ASSERT(service != nullptr);
36 TRPC_ASSERT(codec != nullptr);
37 ServerContextPtr context = MakeRefCounted<ServerContext>();
38
39 context->SetRecvTimestampUs(trpc::time::GetMicroSeconds());
40 context->SetService(service);
41 context->SetFilterController(&(service->GetFilterController()));
42 context->SetServerCodec(codec);
43 context->SetRequestMsg(codec->CreateRequestObject());
44 context->SetResponseMsg(codec->CreateResponseObject());
45
46 return context;
47}
