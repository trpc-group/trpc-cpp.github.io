226 void Swap(MessageBuilder& other) {
227 slice_allocator_.swap(other.slice_allocator_);
228 FlatBufferBuilder::Swap(other);
229 // After swapping the FlatBufferBuilder, we swap back the allocator, which
230 // restores the original allocator back in place. This is necessary because
231 // MessageBuilder's allocator is its own member (SliceAllocatorMember). The
232 // allocator passed to FlatBufferBuilder::vector_downward must point to this
233 // member.
234 buf_.swap_allocator(other.buf_);
235 }

245 template <class T>
246 Message<T> GetMessage() {
247 auto buf_data = buf_.scratch_data(); // pointer to memory
248 auto buf_size = buf_.capacity(); // size of memory
249 auto msg_data = buf_.data(); // pointer to msg
250 auto msg_size = buf_.size(); // size of msg
251 // Do some sanity checks on data/size
252 TRPC_ASSERT(msg_data);
253 TRPC_ASSERT(msg_size);
254 TRPC_ASSERT(msg_data >= buf_data);
255 TRPC_ASSERT(msg_data + msg_size <= buf_data + buf_size);
256 // Calculate offsets from the buffer start
257 auto begin = msg_data - buf_data;
258 auto end = begin + msg_size;
259 // Get the slice we are working with (no refcount change)
260 slice_allocator_.CheckSlice(buf_data, buf_size);
261 TRPC_ASSERT(end >= begin);
262
263 auto subslice = ::trpc::MakeRefCounted<::trpc::Buffer>(msg_size);
264 memcpy(subslice->GetWritePtr(), msg_data, msg_size);
265 subslice->AddWriteLen(msg_size);
266 // Wrap the subslice in a `Message<T>`, but don't increment refcount
267 Message<T> msg(subslice);
268 return msg;
269 }
270
271 template <class T>
272 Message<T> ReleaseMessage() {
273 Message<T> msg = GetMessage<T>();
274 Reset();
275 return msg;
276 }
277};
