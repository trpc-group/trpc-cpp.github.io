105template <typename R, typename T, typename Op, typename InvOp>
106class ReducerSampler : public Sampler {
107 public:
108 using DataType = T;
109 using OpType = Op;
110 static const time_t MAX_SECONDS_LIMIT = 3600;
111
113 explicit ReducerSampler(R* reducer) : reducer_(reducer), window_size_(1) {
114 // Take first sample when initailized.
115 this->TakeSample();
116 }
117
118 ~ReducerSampler() {}
119
121 void TakeSample() override {
122 // window_size may raise.
123 if (static_cast<size_t>(window_size_) + 1 > queue_.Capacity()) {
124 const size_t new_cap = std::max(queue_.Capacity() * 2, (size_t)window_size_ + 1);
125 const size_t mem_size = sizeof(Sample<T>) * new_cap;
126 void* mem = malloc(mem_size);
127 if (mem == nullptr) {
128 return;
129 }
130 trpc::container::BoundedQueue<Sample<T> > new_q(mem, mem_size,
131 trpc::container::StorageOwnership::OWNS_STORAGE);
132 Sample<T> tmp;
133 while (queue_.Pop(&tmp)) {
134 new_q.Push(tmp);
135 }
136 new_q.Swap(queue_);
137 }
138
139 Sample<T> latest;
140 if (is_void_op<InvOp>::value) {
141 latest.data = reducer_->Reset();
142 } else {
143 latest.data = reducer_->GetValue();
144 }
145
146 latest.time_us = GetSystemMicroseconds();
147 queue_.ElimPush(latest);
148 }
149
151 bool GetValue(time_t window_size, Sample<T>* result) const {
152 if (window_size <= 0) {
153 TRPC_FMT_ERROR("Invalid window_size={}", window_size);
154 return false;
155 }
156
157 auto&& const_queue = std::as_const(queue_);
158
159 std::unique_lock<std::mutex> lock(mutex_);
160 if (const_queue.Size() <= 1UL) {
161 return false;
162 }
163 auto* oldest = const_queue.Bottom(window_size);
164 if (oldest == nullptr) {
165 oldest = const_queue.Top();
166 }
167 auto* latest = const_queue.Bottom();
168 TRPC_ASSERT(latest != oldest);
169 if (is_void_op<InvOp>::value) {
170 result->data = latest->data;
171 for (int i = 1; true; ++i) {
172 auto* e = const_queue.Bottom(i);
173 if (e == oldest) {
174 break;
175 }
176 Op()(&result->data, e->data);
177 }
178 } else {
179 result->data = latest->data;
180 InvOp()(&result->data, oldest->data);
181 }
182 result->time_us = latest->time_us - oldest->time_us;
183 return true;
184 }
185
187 int SetWindowSize(time_t window_size) {
188 if (window_size <= 0 || window_size > MAX_SECONDS_LIMIT) {
189 TRPC_FMT_ERROR("Invalid window_size={}", window_size);
190 return -1;
191 }
192
193 std::unique_lock<std::mutex> lock(mutex_);
194 // window_size is not allowed to reduce.
195 if (window_size > window_size_) {
196 window_size_ = window_size;
197 }
198
199 return 0;
200 }
201
203 time_t GetWindowSize() const {
204 std::unique_lock<std::mutex> lock(mutex_);
205 return window_size_;
206 }
207
209 void GetSamples(std::vector<T>* samples, time_t window_size) const {
210 if (window_size <= 0) {
211 TRPC_FMT_ERROR("Invalid window_size={}", window_size);
212 return;
213 }
214
215 auto&& const_queue = std::as_const(queue_);
216
217 std::unique_lock<std::mutex> lock(mutex_);
218 if (const_queue.Size() <= 1) {
219 return;
220 }
221
222 auto* oldest = const_queue.Bottom(window_size);
223 if (oldest == nullptr) {
224 oldest = const_queue.Top();
225 }
226 for (int i = 1; true; ++i) {
227 auto* e = const_queue.Bottom(i);
228 if (e == oldest) {
229 break;
230 }
231 samples->emplace_back(e->data);
232 }
233 }
234
235 private:
236 // Parent containing current reducer sampler.
237 R* reducer_;
238 // In order to limit container size.
239 time_t window_size_;
240 // Container holding datas.
241 trpc::container::BoundedQueue<Sample<T> > queue_;
242};
