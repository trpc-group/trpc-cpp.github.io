49// Similar to `std::data()`, but the return type must be convertible to const char*.
50template <class T>
51constexpr auto data(const T& c)
52 -> std::enable_if_t<std::is_convertible_v<decltype(c.data()), const char*>, const char*> {
53 return c.data();
54}

57constexpr auto data(const char (&c)[N]) {
58 return c;

62constexpr std::size_t size(const T& c) {
63 return c.size();
64}

66// Similar to `std::size` except for it returns number of chars instead of array
67// size (which is 1 greater than number of chars.)
68template <std::size_t N>
69constexpr std::size_t size(const char (&c)[N]) {
70 if (N == 0) {
71 return 0;
72 }
73 return N - (c[N - 1] == 0);

211 void WrapUp(char* ptr, size_t size) {
212 // If `data_` is already set, then `contiguous_buffer_` cannot be managed to avoid unexpected
213 // issues that may arise from mixing them.
214 TRPC_DCHECK(!data_, "contiguous buffer(`ptr`) cannot be mixed with memory blocks in noncontinuous buffers.");
215 // The object only allows the `contiguous_buffer_` to be managed once.
216 TRPC_DCHECK(!contiguous_buffer_, "Only allow to be wrapped up once for `ptr`.");

369 void Skip(std::size_t bytes) noexcept {
370 TRPC_DCHECK_LE(bytes, ByteSize());
371 if (TRPC_UNLIKELY(bytes == 0)) {
372 // NOTHING.
373 } else if (bytes < buffers_.front().size()) {
374 buffers_.front().Skip(bytes);
375 byte_size_ -= bytes;
376 } else {
377 SkipSlow(bytes);

385 NoncontiguousBuffer Cut(std::size_t bytes) {
386 TRPC_DCHECK_LE(bytes, ByteSize());
387
388 if (bytes == ByteSize()) {
389 return std::move(*this);
390 }
391
392 NoncontiguousBuffer rc;
393 auto left = bytes;
394
395 while (left && left >= buffers_.front().size()) {
396 left -= buffers_.front().size();
397 rc.buffers_.push_back(buffers_.pop_front());
398 }
399
400 if (TRPC_LIKELY(left)) {
401 BufferBlock* ncb = object_pool::MakeLwUnique<BufferBlock>().Leak();
402 *ncb = buffers_.front();
403 ncb->SetSize(left);
404 rc.buffers_.push_back(ncb);
405 buffers_.front().Skip(left);
406 }
407 rc.byte_size_ = bytes;
408 byte_size_ -= bytes;
409 return rc;
410 }

514class NoncontiguousBufferBuilder {
515 public:
516 NoncontiguousBufferBuilder() { InitializeNextBlock(); }

520 char* data() const noexcept { return current_->data + used_; }

524 std::size_t SizeAvailable() const noexcept { return GetBlockMaxAvailableSize() - used_; }

529 void MarkWritten(std::size_t bytes) {
530 TRPC_DCHECK_LE(bytes, SizeAvailable(), "You're overflowing the buffer.");
531 used_ += bytes;
532 if (TRPC_UNLIKELY(!SizeAvailable())) {
533 FlushCurrentBlock();
534 InitializeNextBlock();
535 }
536 }
537
542 char* Reserve(std::size_t bytes) {
543 const std::size_t kMaxBytes = GetBlockMaxAvailableSize();
544
545 TRPC_CHECK_LE(bytes, kMaxBytes, "At most [{}] bytes may be reserved in a single call.", kMaxBytes);
546 if (SizeAvailable() < bytes) {
547 // There is not enough space available in the intermediate contiguous buffer, so a new one needs to be created.
548 FlushCurrentBlock();
549 InitializeNextBlock();
550 }
551 auto* ptr = data();
552 MarkWritten(bytes);
553 return ptr;
554 }
555
558 std::size_t ByteSize() const noexcept { return nb_.ByteSize() + used_; }
559
563 NoncontiguousBuffer DestructiveGet() noexcept {
564 FlushCurrentBlock();
565 return std::move(nb_);
566 }
567
573 void Append(const void* ptr, std::size_t length) {
574 if (length == 0) {
575 return;
576 }
577
578 auto current = data();
579 // First, increase the value of `used_`. This operation may cause `used_` to temporarily overflow.
580 // If it overflows, use the `AppendSlow` method to continue the operation.
581 used_ += length;
582 if (TRPC_LIKELY(used_ < GetBlockMaxAvailableSize())) {
583 // If the current size of the intermediate contiguous buffer is sufficient, simply perform a direct copy.
584#if __GNUC__ == 10
585#pragma GCC diagnostic push
586#pragma GCC diagnostic ignored "-Wstringop-overflow"
587#endif
588 memcpy(current, ptr, length);
589#if __GNUC__ == 10
590#pragma GCC diagnostic pop
591#endif
592 return;
593 }
594 used_ -= length; // If the available space is not enough, you will need to use a slower copying method.
595 AppendSlow(ptr, length);
596 }

600 void Append(const std::string_view& s) { return Append(s.data(), s.size()); }
601
604 void Append(BufferView v) { return Append(v.data(), v.size()); }

608 void Append(NoncontiguousBuffer buffer) {
609 // If there have been no operations on the intermediate contiguous cache, there is no need to execute this step.
610 if (used_) {
611 FlushCurrentBlock();
612 InitializeNextBlock();
613 }
614 nb_.Append(std::move(buffer));
615 }

632 template <class... Ts, class = std::void_t<decltype(detail::data(std::declval<Ts&>()))...>,
633 class = std::void_t<decltype(detail::size(std::declval<Ts&>()))...>,
634 class = std::enable_if_t<(sizeof...(Ts) > 1)>>
635 void Append(const Ts&... buffers) {
636 auto current = data();
637 auto total = (detail::size(buffers) + ...);
638 used_ += total;
639 if (TRPC_LIKELY(used_ < GetBlockMaxAvailableSize())) {
640 UncheckedAppend(current, buffers...);
641 return;
642 }
643
644 used_ -= total;
645 [[maybe_unused]] int dummy[] = {(Append(buffers), 0)...};
646 }
647
648 private:
649 // Allocate a new contiguous buffer.
650 void InitializeNextBlock();
651
652 // Move the currently used contiguous buffer to NoncontiguousBuffer.
653 void FlushCurrentBlock();
654
655 void AppendSlow(const void* ptr, std::size_t length);

657 template <class T, class... Ts>
658 [[gnu::always_inline]] void UncheckedAppend(char* ptr, const T& sv) {
659 memcpy(ptr, detail::data(sv), detail::size(sv));
660 }

663 [[gnu::always_inline]] void UncheckedAppend(char* ptr, const T& sv, const Ts&... svs) {
664 memcpy(ptr, detail::data(sv), detail::size(sv));
665 UncheckedAppend(ptr + detail::size(sv), svs...);
666 }
