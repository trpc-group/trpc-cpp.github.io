78template <class T, class>
79struct TryParseTraits {
80 // The default implementation delegates calls to `TryParse` found by ADL.
81 template <class... Args>
82 static std::optional<T> TryParse(const std::string_view& s, const Args&... args) {
83 return TryParse(std::common_type<T>(), s, args...);
84 }
85};

87template <>
88struct TryParseTraits<bool> {
89 // For numerical values, only 0 and 1 are recognized, all other numeric values
90 // lead to parse failure (i.e., `std::nullopt` is returned).
91 //
92 // If `recognizes_alphabet_symbol` is set, following symbols are recognized:
93 //
94 // - "true" / "false"
95 // - "y" / "n"
96 // - "yes" / "no"
97 //
98 // For all other symbols, we treat them as neither `true` nor `false`.
99 // `std::nullopt` is returned in those cases.
100 static std::optional<bool> TryParse(const std::string_view& s, bool recognizes_alphabet_symbol = true,
101 bool ignore_case = true);
102};
