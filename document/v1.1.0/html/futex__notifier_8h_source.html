64 bool Wait(const State& expected_state, const std::chrono::steady_clock::time_point* abs_time) {
65 timespec timeout;
66 timespec* timeout_ptr = nullptr;
67
68 while (true) {
69 if (abs_time != nullptr) {
70 auto now = ReadSteadyClock();
71
72 // Already timeout.
73 if (*abs_time <= now)
74 return false;
75
76 auto diff = (*abs_time - now) / std::chrono::nanoseconds(1);
77 // Convert format.
78 timeout.tv_sec = diff / 1000000000L;
79 timeout.tv_nsec = diff - timeout.tv_sec * 1000000000L;
80 timeout_ptr = &timeout;
81 }
82
83 int ret = syscall(SYS_futex, &pending_wake_, (FUTEX_WAIT | FUTEX_PRIVATE_FLAG),
84 expected_state.val_, timeout_ptr, nullptr, 0);
85 // Timeout occured.
86 if ((ret != 0) && (errno == ETIMEDOUT))
87 return false;
88
89 // The value pointed to by uaddr was not equal to the expected value val at the time of the call.
90 // Take it as awakened up by another thread already.
91 if ((ret != 0) && (errno == EAGAIN))
92 return true;
93
94 // Interrupted, just continue to guarantee enough timeout.
95 if ((ret != 0) && (errno == EINTR))
96 continue;
97
98 // Spurious wake-up, just continue.
99 if ((ret == 0) && (pending_wake_ == expected_state.val_))
100 continue;
101
102 // Awakened up by others or real timeout.
103 return (ret == 0) ? true : false;
104 }
105 }
106
107 void Stop() {
108 pending_wake_.fetch_or(1);
109 syscall(SYS_futex, &pending_wake_, (FUTEX_WAKE | FUTEX_PRIVATE_FLAG),
110 10000, NULL, NULL, 0);
111 }
112
113 private:
114 std::atomic<int> pending_wake_;
115};
