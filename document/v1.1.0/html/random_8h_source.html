38inline std::mt19937_64& Engine() {
39 // This could be quite tricky.
40 //
41 // Ideally we'd like to initialize the PRNG engine's internal state with raw
42 // bits from `std::random_device`. However, per spec, the engine only accepts
43 // a single number, or a `SeedSequence`, which is not satisfied by
44 // `std::random-device`, in its constructor. In the meantime, by using
45 // `std::seed_seq`, we potentially degrade our randomness.
46 //
47 // It would be the best if the Standard requires `std::random_device` to
48 // satisfy `SeedSequence`, or (better yet), allows `std::mt19937` to accept a
49 // `std::random_device`.
50 //
51 // More detailed discussion about the initialization of PRNG engine could be
52 // found at:
53 //
54 // - https://stackoverflow.com/questions/15509270
55 // - http://www.pcg-random.org/posts/cpp-seeding-surprises.html
56 // - https://www.reddit.com/r/cpp/comments/31857s
57 //
58 // Anyway, below is a more-or-less acceptable initialization, per the SO
59 // answer linked above.
60 thread_local auto engine = [] {
61 std::random_device r;
62 std::seed_seq seeds{r(), r(), r(), r(), r(), r(), r(), r()};
63 // Use unique_ptr to avoid much space in TLS.
64 auto engine = std::make_unique<std::mt19937_64>(seeds);
65 return engine;
66 }();
67 return *engine;
68}

74template <class T = std::uint64_t, class = std::enable_if_t<std::is_integral_v<T>>>
75T Random() {

85template <class T>
86T Random(T min, T max) {
87 using Distribution = std::conditional_t<std::is_floating_point_v<T>, std::uniform_real_distribution<T>,
88 std::uniform_int_distribution<T>>;
89 Distribution dist(min, max); // May NOT be `thread_local` as `min` / `max`
90 // could be different across calls.
91 return dist(detail::Engine());
92}
