46 template <class RequestProtocol>
47 void SetRequest(const ClientContextPtr& context, const RequestProtocol& req) {
48 // Set req parameter to context. If req parameter is equal to req in context, it will do nothing.
49 const ProtocolPtr& req_protocol = context->GetRequest();
50 if (req_protocol == nullptr) {
51 context->SetRequest(req);
52 } else if (TRPC_UNLIKELY(req_protocol.get() != req.get())) {
53 // If the req in the context isn't empty and is not equal to the input parameter, print a log message indicating
54 // that it has been overwritten
55 TRPC_FMT_WARN_EVERY_SECOND("`req` will replace of `request of context`");
56 context->SetRequest(req);
57 }
58 }

61template <class RequestProtocol, class ResponseProtocol>
62Status NonRpcServiceProxy::UnaryInvoke(const ClientContextPtr& context, const RequestProtocol& req,
63 ResponseProtocol& rsp) {
64 const bool is_protocol_req = std::is_convertible_v<typename RequestProtocol::element_type*, Protocol*>;
65 const bool is_protocol_rsp = std::is_convertible_v<typename ResponseProtocol::element_type*, Protocol*>;
66
67 static_assert(is_protocol_req, "RequestProtocol::element_type is not derive of 'Protocol'");
68 static_assert(is_protocol_rsp, "ResponseProtocol::element_type is not derive of 'Protocol'");
69
70 SetRequest(context, req);
71
72 FillClientContext(context);
73
74 auto filter_ret = RunFilters(FilterPoint::CLIENT_PRE_RPC_INVOKE, context);
75 if (filter_ret == 0) {
76 ProtocolPtr& rsp_protocol = context->GetResponse();
77 Status unary_invoke_status = ServiceProxy::UnaryInvoke(context, req, rsp_protocol);
78 if (unary_invoke_status.OK()) {
79 rsp = std::dynamic_pointer_cast<typename ResponseProtocol::element_type>(rsp_protocol);
80 }
81 }
82
83 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
84
85 context->SetEndTimestampUs(trpc::time::GetMicroSeconds());
86
87 return context->GetStatus();
88}

90template <class RequestProtocol, class ResponseProtocol>
91Future<ResponseProtocol> NonRpcServiceProxy::AsyncUnaryInvoke(const ClientContextPtr& context,
92 const RequestProtocol& req) {
93 const bool is_protocol_req = std::is_convertible_v<typename RequestProtocol::element_type*, Protocol*>;
94 const bool is_protocol_rsp = std::is_convertible_v<typename ResponseProtocol::element_type*, Protocol*>;
95
96 static_assert(is_protocol_req, "RequestProtocol::element_type is not derive of 'Protocol'");
97 static_assert(is_protocol_rsp, "ResponseProtocol::element_type is not derive of 'Protocol'");
98
99 SetRequest(context, req);
100
101 FillClientContext(context);
102
103 auto filter_ret = RunFilters(FilterPoint::CLIENT_PRE_RPC_INVOKE, context);
104 if (filter_ret != 0) {
105 TRPC_FMT_ERROR("service name:{}, filter execute failed.", GetServiceName());
106 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
107
108 context->SetEndTimestampUs(trpc::time::GetMicroSeconds());
109
110 const Status& result = context->GetStatus();
111 return MakeExceptionFuture<ResponseProtocol>(CommonException(result.ErrorMessage().c_str()));
112 }
113
114 return ServiceProxy::AsyncUnaryInvoke(context, req).Then([this, context](Future<ProtocolPtr>&& rsp) {
115 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
116
117 context->SetEndTimestampUs(trpc::time::GetMicroSeconds());
118
119 if (rsp.IsReady()) {
120 ProtocolPtr result = rsp.GetValue0();
121 ResponseProtocol rsp_data = std::dynamic_pointer_cast<typename ResponseProtocol::element_type>(result);
122 return MakeReadyFuture<ResponseProtocol>(std::move(rsp_data));
123 } else {
124 return MakeExceptionFuture<ResponseProtocol>(rsp.GetException());
125 }
126 });
127}

129template <class RequestProtocol>
130Status NonRpcServiceProxy::OnewayInvoke(const ClientContextPtr& context, const RequestProtocol& req) {
131 const bool is_protocol_req = std::is_convertible_v<RequestProtocol, ProtocolPtr>;
132
133 static_assert(is_protocol_req, "RequestProtocol is not derive of 'ProtocolPtr'");
134
135 SetRequest(context, req);
136
137 FillClientContext(context);
138
139 context->SetCallType(kOnewayCall);
140
141 auto filter_ret = RunFilters(FilterPoint::CLIENT_PRE_RPC_INVOKE, context);
142 if (filter_ret == 0) {
143 ServiceProxy::OnewayInvoke(context, req);
144 }
145 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
146
147 context->SetEndTimestampUs(trpc::time::GetMicroSeconds());
148
149 return context->GetStatus();
150}
