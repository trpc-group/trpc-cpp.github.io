29template <typename T>
30class alignas(64) BoundedMPMCQueue {
31 public:
32 static_assert(std::is_default_constructible_v<T>, "Type must be constructible");
33 static_assert(std::is_move_constructible<T>::value, "Types must be move constructible");
34
36 constexpr static int kRetOk = 0;
38 constexpr static int kRetFull = 1;
40 constexpr static int kRetEmpty = 2;
42 constexpr static int kRetError = -1;
43
44 BoundedMPMCQueue() = default;
45
48 bool Init(size_t size) {
49 capacity_ = RoundUpPowerOf2(size);
50
51 elements_ = std::make_unique<Element[]>(capacity_);
52 for (size_t i = 0; i < capacity_; ++i) {
53 elements_[i].sequence = i;
54 }
55
56 enqueue_pos_.store(0, std::memory_order_relaxed);
57 dequeue_pos_.store(0, std::memory_order_relaxed);
58
59 return true;
60 }
61
65 bool Push(T data) {
66 Element* element = nullptr;
67 size_t pos = 0;
68
69 while (true) {
70 pos = enqueue_pos_.load(std::memory_order_relaxed);
71 element = &elements_[pos & (capacity_ - 1)];
72 size_t seq = element->sequence.load(std::memory_order_acquire);
73 intptr_t dif = (intptr_t)seq - (intptr_t)pos;
74
75 if (dif == 0) {
76 if (enqueue_pos_.compare_exchange_weak(pos, pos + 1, std::memory_order_relaxed)) {
77 break;
78 }
79 } else if (dif < 0) {
80 if ((pos - dequeue_pos_.load(std::memory_order_relaxed)) == capacity_) {
81 return false;
82 }
83 }
84 queue::detail::Pause();
85 }
86
87 element->data = std::move(data);
88 element->sequence.store(pos + 1, std::memory_order_release);
89
90 return true;
91 }
92
96 bool Pop(T& data) {
97 return Dequeue(&data) == kRetOk;
98 }

106 int Enqueue(T&& data, bool block = false) {
107 Element* elem = nullptr;
108 size_t pos = 0;
109
110 while (true) {
111 pos = enqueue_pos_.load(std::memory_order_relaxed);
112 elem = &elements_[pos & (capacity_ - 1)];
113 size_t seq = elem->sequence.load(std::memory_order_acquire);
114 intptr_t dif = (intptr_t)seq - (intptr_t)pos;
115
116 if (dif == 0) {
117 if (enqueue_pos_.compare_exchange_weak(pos, pos + 1, std::memory_order_relaxed)) {
118 break;
119 }
120 } else if (dif < 0) {
121 if ((pos - dequeue_pos_.load(std::memory_order_relaxed)) == capacity_) {
122 if (!block) {
123 return kRetFull;
124 }
125 }
126 }
127
128 queue::detail::Pause();
129 }
130
131 elem->data = std::move(data);
132 elem->sequence.store(pos + 1, std::memory_order_release);
133
134 return kRetOk;
135 }

143 int Dequeue(T* data, bool block = false) {
144 Element* elem = nullptr;
145 size_t pos = 0;
146
147 while (true) {
148 pos = dequeue_pos_.load(std::memory_order_relaxed);
149 elem = &elements_[pos & (capacity_ - 1)];
150 size_t seq = elem->sequence.load(std::memory_order_acquire);
151 intptr_t dif = (intptr_t)seq - (intptr_t)(pos + 1);
152
153 if (dif == 0) {
154 if (dequeue_pos_.compare_exchange_weak(pos, pos + 1, std::memory_order_relaxed)) {
155 break;
156 }
157 } else if (dif < 0) {
158 if (pos - enqueue_pos_.load(std::memory_order_relaxed) == 0) {
159 if (!block) {
160 return kRetEmpty;
161 }
162 }
163 }
164
165 queue::detail::Pause();
166 }
167
168 *data = std::move(elem->data);
169 elem->sequence.store(pos + capacity_, std::memory_order_release);
170
171 return kRetOk;
172 }

175 uint32_t Size() const {
176 size_t enqueue_pos = enqueue_pos_.load(std::memory_order_relaxed);
177 size_t dequeue_pos = dequeue_pos_.load(std::memory_order_relaxed);
178 return (enqueue_pos <= dequeue_pos ? 0 : (enqueue_pos - dequeue_pos));
179 }

182 uint32_t Capacity() const { return static_cast<uint32_t>(capacity_); }
