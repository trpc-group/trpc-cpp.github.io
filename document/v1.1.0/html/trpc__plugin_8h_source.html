171 template <typename T>
172 bool AddPlugins(const std::unordered_map<std::string, T>& plugins) {
173 for (const auto& it : plugins) {
174 PluginInfo plugin_info;
175 plugin_info.plugin = it.second;
176 plugin_info.is_inited = false;
177 plugin_info.is_destroyed = false;
178
179 std::string plugin_name = it.second->Name() + "#" + std::to_string(static_cast<int>(it.second->Type()));
180 plugins_.emplace(plugin_name, std::move(plugin_info));
181
182 std::vector<std::string> dep_plugin_names;
183 it.second->GetDependencies(dep_plugin_names);
184 // Check if the dependent plugin name is valid
185 TRPC_ASSERT(IsDepPluginNameValid(dep_plugin_names) && "Dependent plugin name invalid");
186 for (auto& dep_plugin_name : dep_plugin_names) {
187 plugins_reverse_deps_[dep_plugin_name].push_back(plugin_name);
188 }
189 }
190
191 return true;
192 }
