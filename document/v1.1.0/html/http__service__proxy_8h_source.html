329 ConstructHttpRequestHeader(context, OperationType::POST, url);

371template <class RequestMessage, class ResponseMessage>
372Status HttpServiceProxy::HttpUnaryInvoke(const ClientContextPtr& context, const RequestMessage& req,
373 ResponseMessage* rsp) {
374 if (!context->GetRequest()) {
375 context->SetRequest(codec_->CreateRequestPtr());
376 }
377
378 ProtocolPtr& req_protocol = context->GetRequest();
379 auto* req_msg = dynamic_cast<HttpRequestProtocol*>(req_protocol.get());
380 // The conversion failed because `req_protocol` was created by the wrong codec.
381 if (TRPC_UNLIKELY(!req_msg)) {
382 return Status(static_cast<int>(codec::ClientRetCode::ENCODE_ERROR), "unmatched codec");
383 }
384 *req_msg->request = req;
385
386 // Pass the request method to the transport. Response content is not required for HTTP HEAD request.
387 context->SetCurrentContextExt(static_cast<uint32_t>(req_msg->request->GetMethodType()));
388
389 FillClientContext(context);
390
391 ProtocolPtr& rsp_protocol = context->GetResponse();
392
393 InnerUnaryInvoke(context, req_protocol, rsp_protocol);
394
395 if (context->GetStatus().OK()) {
396 auto* ret_rsp = static_cast<HttpResponseProtocol*>(rsp_protocol.get());
397 *rsp = std::move(ret_rsp->response);
398 }
399 return context->GetStatus();
400}

402template <class RequestMessage, class ResponseMessage>
403Future<ResponseMessage> HttpServiceProxy::AsyncHttpUnaryInvoke(const ClientContextPtr& context,
404 const RequestMessage& req) {
405 if (!context->GetRequest()) {
406 context->SetRequest(codec_->CreateRequestPtr());
407 }
408
409 ProtocolPtr& req_protocol = context->GetRequest();
410 auto* req_msg = dynamic_cast<HttpRequestProtocol*>(req_protocol.get());
411 // The conversion failed because `req_protocol` was created by the wrong codec.
412 if (TRPC_UNLIKELY(!req_msg)) {
413 return MakeExceptionFuture<ResponseMessage>(CommonException("unmatched codec"));
414 }
415 *req_msg->request = req;
416
417 // Pass the request method to the transport. Response content is not required for HTTP HEAD request.
418 context->SetCurrentContextExt(static_cast<uint32_t>(req_msg->request->GetMethodType()));
419
420 FillClientContext(context);
421
422 return AsyncInnerUnaryInvoke(context, req_protocol).Then([context](Future<ProtocolPtr>&& rsp_fut) {
423 if (!rsp_fut.IsFailed()) {
424 context->SetResponse(rsp_fut.GetValue0());
425
426 ProtocolPtr& rsp_protocol = context->GetResponse();
427 auto* rsp = static_cast<HttpResponseProtocol*>(rsp_protocol.get());
428 return MakeReadyFuture<ResponseMessage>(std::move(rsp->response));
429 }
430 return MakeExceptionFuture<ResponseMessage>(rsp_fut.GetException());
431 });
432}
