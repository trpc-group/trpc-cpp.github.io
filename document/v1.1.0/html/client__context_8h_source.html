802 struct alignas(8) InvokeInfo {
803 // Unique ID of request.
804 uint32_t seq_id = 0;
805
806 // Stream id.
807 uint32_t stream_id = 0;
808
809 // Timeout (ms) for requesting.
810 uint32_t timeout = UINT32_MAX;
811
812 // Type of RPC call.
813 uint8_t call_type = TRPC_UNARY_CALL;
814
815 // Bit flags for context state, the bit flag from low to high is as following:
816 // 1: kIsUseFullLinkTimeoutMask, indicates whether full-link timeout is used by user.
817 // 2: kIsSetAddrMask, indicates whether IP:Port is set by user.
818 // 3: kIsTransparentMask, indicates whether it is a transparent forwarding request.
819 // 4: kIsFixedConnectorMask, indicates whether request will be sending over fixed connector.
820 // 5: kIsBackupRequestMask, indicates whether backup-request is used by user.
821 // 6: kIsIgnoreProxyTimeoutMask, indicates whether to ignore timeout option of proxy.
822 // 7: kIsSetRequestId, used to indicate whether the request ID has been set.
823 // 8: reserved.
824 uint8_t state_flag_ = 0b00000000;
825
826 // Type of message.
827 uint32_t message_type{0};
828
829 // Type of content-encoding for request message, protobuf is the default (value: 0).
830 uint8_t req_encode_type{0};
831
832 // Type of content-encoding for request message object, protobuf message is the default (value: 0).
833 // The framework will select the corresponding content-encoding-type according to the request
834 // message type. The value set by user will be ignored.
835 uint8_t req_encode_data_type{0};
836
837 // Type of content-encoding for response message, protobuf is the default (value: 0).
838 uint8_t rsp_encode_type{0};
839
840 // Type of content-encoding for request message object, protobuf message is the default.
841 // The framework will select the corresponding content-encoding-type according to the response
842 // message type. the settings set by user will be ignored.
843 uint8_t rsp_encode_data_type{0};
844
845 // Compression settings of request: compression type (high 8 bits) + compression level (low 8 bits).
846 // NONE:DEFAULT is the default.
847 uint16_t req_compress_info{static_cast<uint16_t>(compressor::kNone << 8) | compressor::kDefault};
848
849 // Compression settings of response: compression type (high 8 bits) + compression level (low 8 bits).
850 // NONE:DEFAULT is the default.
851 uint16_t rsp_compress_info{static_cast<uint16_t>(compressor::kNone << 8) | compressor::kDefault};
852
853 // The function name of the caller who issues an RPC call.
854 std::string caller_func_name;
855
856 // Method descriptor used internally within protobuf.
857 const google::protobuf::MethodDescriptor* method_desc{nullptr};
858 };
