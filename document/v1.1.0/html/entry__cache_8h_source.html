38struct EntryCache {
39 // Note that `nullptr` would cause both `Get()` and `Put()` to fall to slow
40 // path, which is desired.
41 Entry** current = nullptr;
42 Entry** bottom = nullptr;
43 Entry** top = nullptr;
44
47 Entry* Get() noexcept {
48 if (TRPC_LIKELY(current > bottom)) {
49 return *--current;
50 }
51 return GetSlow();
52 }
53
56 void Put(Entry* entry) noexcept {
57 if (TRPC_LIKELY(current < top)) {
58 *current++ = entry;
59 return;
60 }
61 PutSlow(entry);
62 }
63
64 private:
65 void InitializeOnceCheck();
66
67 Entry* GetSlow() noexcept;
68 void PutSlow(Entry* entry) noexcept;
69};
