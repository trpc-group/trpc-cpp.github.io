19class ErasedPtr {
20 public:
21 using Deleter = void (*)(void*);
22
23 // A default constructed one is an empty one.
24 // implicit
25 constexpr ErasedPtr(std::nullptr_t = nullptr) // NOLINT
26 : ptr_(nullptr), deleter_(nullptr) {}
27
28 // Ownership taken.
29 template <class T>
30 constexpr explicit ErasedPtr(T* ptr) noexcept : ptr_(ptr), deleter_([](void* ptr) { delete static_cast<T*>(ptr); }) {}
31 template <class T, class D>
32 constexpr ErasedPtr(T* ptr, D deleter) noexcept : ptr_(ptr), deleter_(deleter) {}
33
34 // Movable
35 constexpr ErasedPtr(ErasedPtr&& ptr) noexcept : ptr_(ptr.ptr_), deleter_(ptr.deleter_) { ptr.ptr_ = nullptr; }
36
37 ErasedPtr& operator=(ErasedPtr&& ptr) noexcept {
38 if (&ptr != this) {
39 Reset();
40 }
41 std::swap(ptr_, ptr.ptr_);
42 std::swap(deleter_, ptr.deleter_);
43 return *this;
44 }
45
46 // But not copyable.
47 ErasedPtr(const ErasedPtr&) = delete;
48 ErasedPtr& operator=(const ErasedPtr&) = delete;
49
50 // Any resource we're holding is freed in dtor.
51 ~ErasedPtr() {
52 if (ptr_) {
53 deleter_(ptr_);
54 }
55 }
56
59 constexpr void* Get() const noexcept { return ptr_; }
60
63 template <class T>
64 T* UnsafeGet() const noexcept { // Or `UncheckedGet()`?
65 return reinterpret_cast<T*>(Get());
66 }
67
69 constexpr explicit operator bool() const noexcept { return !!ptr_; }
70
72 constexpr void Reset(std::nullptr_t = nullptr) noexcept {
73 if (ptr_) {
74 deleter_(ptr_);
75 ptr_ = nullptr;
76 }
77 }
78
80 [[nodiscard]] void* Leak() noexcept { return std::exchange(ptr_, nullptr); }
81
83 constexpr Deleter GetDeleter() const noexcept { return deleter_; }
84
85 private:
86 void* ptr_;
87 Deleter deleter_;
88};

92ErasedPtr MakeErased(Args&&... args) {

94}
