77 template <typename Clock, typename Dur>
78 Status ReadHeaders(int& code, trpc::http::HttpHeader& http_header,
79 const std::chrono::time_point<Clock, Dur>& expiry) {
80 std::unique_lock<FiberMutex> lk(http_response_mutex_);
81 if (!http_response_can_read_.wait_until(lk, expiry, [this] { return http_response_ || (state_ & kClosed); })) {
82 return kStreamStatusClientReadTimeout;
83 }
84
85 if (http_response_) {
86 code = http_response_->GetStatus();
87 http_header = http_response_->GetHeader();
88 return kDefaultStatus;
89 }
90 return kStreamStatusClientNetworkError;
91 }

106 template <typename Clock, typename Dur>
107 Status Read(NoncontiguousBuffer& item, size_t max_bytes, const std::chrono::time_point<Clock, Dur>& expiry) {
108 if ((state_ & (kClosed | kReading)) == (kClosed | kReading)) { // closed while reading
109 return kStreamStatusClientNetworkError;
110 }
111
112 if (state_ & kReadEof) {
113 return kStreamStatusReadEof;
114 }
115
116 if (!read_buffer_.Cut(item, max_bytes, expiry)) {
117 return kStreamStatusClientReadTimeout;
118 }
119
120 // The end of the `Cut` operation may be due to a normal return, or it may be due to an interruption from
121 // `Close/EOF`.
122 if (state_ & kClosed) {
123 return kStreamStatusClientNetworkError;
124 }
125
126 if (item.ByteSize() < max_bytes) {
127 // The first occurrence of `EOF` is considered to be normal, and only subsequent retries to read will return
128 // `kStreamStatusReadEof`.
129 state_ |= kReadEof;
130 }
131
132 return kDefaultStatus;
133 }
