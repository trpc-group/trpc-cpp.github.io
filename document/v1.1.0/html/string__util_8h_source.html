62template <typename T = std::string>
63std::vector<T> SplitString(std::string_view in, char separator) {
64 std::vector<T> output;
65 auto begin = in.begin(), end = in.end();
66 auto first = begin;
67
68 while (first != end) {
69 auto second = std::find(first, end, separator);
70 if (first != second) output.emplace_back(in.substr(std::distance(begin, first), std::distance(first, second)));
71 if (second == end) break;
72 first = ++second;
73 }
74
75 return output;
76}

134template <class T>
135std::string Join(const std::vector<T>& in, const std::string& sep) {
136 std::stringstream stream;
137
138 bool first = true;
139 for (const auto& s : in) {
140 if (first)
141 first = false;
142 else
143 stream << sep;
144
145 stream << s;
146 }
147
148 return stream.str();
149}
