51#define TRPC_LOG(instance, level, msg) \
52 do { \
53 const auto& __TRPC_LOG_INSTANCE__ = ::trpc::LogFactory::GetInstance()->Get(); \
54 if (__TRPC_LOG_INSTANCE__) { \
55 if (__TRPC_LOG_INSTANCE__->ShouldLog(instance, level)) \
56 __TRPC_LOG_INSTANCE__->LogIt(instance, level, __FILE__, __LINE__, __FUNCTION__, msg); \
57 } else { \
58 ::trpc::Log::NoLog(instance, level, __FILE__, __LINE__, __FUNCTION__, msg); \
59 } \
60 } while (0)

297#define TRPC_ONCE() \
298 ({ \
299 static std::atomic_flag __TRPC_LOG_FLAG__ = ATOMIC_FLAG_INIT; \
300 (!__TRPC_LOG_FLAG__.test_and_set(std::memory_order_relaxed)); \
301 })

306#define TRPC_EVERY_N(n) \
307 ({ \
308 TRPC_ASSERT((n) > 0 && "should not be less than 1"); \
309 static std::atomic<uint64_t> __TRPC_LOG_OCCURENCE_N__{(n)}; \
310 (__TRPC_LOG_OCCURENCE_N__.fetch_add(1, std::memory_order_relaxed) % (n) == 0); \
311 })

316#define TRPC_WITHIN_N(ms) \
317 ({ \
318 TRPC_ASSERT((ms) > 0 && "should not be less than 1"); \
319 bool __TRPC_LOG_WRITEABLE__ = false; \
320 static std::chrono::time_point<std::chrono::steady_clock> __TRPC_TIME_POINT__; \
321 static std::atomic_flag __TRPC_LOG_CONDITION_FLAG__ = ATOMIC_FLAG_INIT; \
322 if (!__TRPC_LOG_CONDITION_FLAG__.test_and_set(std::memory_order_acquire)) { \
323 auto __TRPC_NOWS__ = std::chrono::steady_clock::now(); \
324 int64_t __TRPC_DIFF_N__ = \
325 std::chrono::duration_cast<std::chrono::milliseconds>(__TRPC_NOWS__ - __TRPC_TIME_POINT__).count(); \
326 if (__TRPC_DIFF_N__ > ms) { \
327 __TRPC_TIME_POINT__ = __TRPC_NOWS__; \
328 __TRPC_LOG_WRITEABLE__ = true; \
329 } \
330 __TRPC_LOG_CONDITION_FLAG__.clear(std::memory_order_release); \
331 } \
332 __TRPC_LOG_WRITEABLE__; \
333 })
