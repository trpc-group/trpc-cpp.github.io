32struct IdentityTime {
33 uint64_t operator()(const uint64_t& val) const { return val; }

36struct TscToDuration {
37 std::uint64_t operator()(const uint64_t& val) const {
38 return DurationFromTsc(0, val) / std::chrono::nanoseconds(1);
39 }

42template <typename F = IdentityTime>
43class InternalLatency {
44 public:
45 explicit InternalLatency(std::string_view rel_path) {
46 LinkToParent(rel_path, TrpcVarGroup::FindOrCreate("/"));
47 }
48
49 InternalLatency(TrpcVarGroup* parent, std::string_view rel_path) {
50 LinkToParent(rel_path, parent);
51 }
52
53 void Update(uint64_t v) {
54 min_.Update(v);
55 max_.Update(v);
56 sum_.Add(v);
57 cnt_.Increment();
58 }
59
60 bool IsExposed() const { return handle_.has_value(); }
61
62 private:
63 Json::Value ToJsonValue(const InternalLatency<F>& v) {
64 Json::Value result;
65 {
66 auto cnt = v.cnt_.GetValue();
67 auto sum = v.sum_.GetValue();
68 result["total"]["avg"] = detail::ToJsonValue(F()(cnt != 0 ? (sum / cnt) : cnt));
69 result["total"]["cnt"] = static_cast<Json::UInt64>(cnt);
70 }
71 static const std::vector<std::pair<std::string, time_t>> m = {
72 {"1s", 1}, {"1min", 60}, {"10min", 600}, {"30min", 1800}, {"1h", 3600}};
73 for (auto&& [k, r] : m) {
74 auto cnt = v.win_cnt_.GetValue(r);
75 auto sum = v.win_sum_.GetValue(r);
76 result[k]["avg"] = detail::ToJsonValue(F()(cnt != 0 ? (sum / cnt) : cnt));
77 result[k]["min"] = detail::ToJsonValue(F()(v.win_min_.GetValue(r)));
78 result[k]["max"] = detail::ToJsonValue(F()(v.win_max_.GetValue(r)));
79 result[k]["cnt"] = static_cast<Json::UInt64>(cnt);
80 }
81 return result;
82 }
83
84 void LinkToParent(std::string_view rel_path, TrpcVarGroup* parent) {
85 handle_ = parent->Add(rel_path, [this] { return ToJsonValue(*this); });
86 }
87
88 private:
89 Miner<uint64_t> min_{};
90 Maxer<uint64_t> max_{};
91 Counter<uint64_t> cnt_{};
92 Counter<uint64_t> sum_{};
93 Window<Miner<uint64_t>> win_min_{&min_, 3600};
94 Window<Maxer<uint64_t>> win_max_{&max_, 3600};
95 Window<Counter<uint64_t>> win_cnt_{&cnt_, 3600};
96 Window<Counter<uint64_t>> win_sum_{&sum_, 3600};
97
98 std::optional<TrpcVarGroup::Handle> handle_;
99};
