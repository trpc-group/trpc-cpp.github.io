79constexpr compressor::CompressType StringToCompressType(std::string_view type) {
80 // 1. Currently only supports zlib/gzip/snappy decompression, where snappy is an extension.
81 // 2. Currently only supports applying one compression algorithm, does not support applying two compression
82 // algorithms in succession, and does not support applying zlib first and then applying gzip
83 // (Content-Encoding: deflate, gzip).
84 if (type == "gzip") {
85 return compressor::kGzip;
86 } else if (type == "deflate") {
87 return compressor::kZlib;
88 } else if (type == "snappy") {
89 return compressor::kSnappy;
90 } else if (type == "identity") {
91 return compressor::kNone;
92 }
93 return compressor::kMaxType;
94}
