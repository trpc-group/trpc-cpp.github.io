59class PredicateNotifier {
60 public:
62 struct Waiter {
63 std::atomic<Waiter*> next;
64 std::mutex mu;
65 std::condition_variable cv;
66 uint64_t epoch;
67 unsigned state;
68 enum {
69 kNotSignaled,
70 kWaiting,
71 kSignaled,
72 };
73 };
74
75 explicit PredicateNotifier(size_t N) : waiters_{N} {
76 assert(waiters_.size() < (1 << kWaiterBits) - 1);
77 // Initialize epoch to something close to overflow to test overflow.
78 state_ = kStackMask | (kEpochMask - kEpochInc * waiters_.size() * 2);
79 }
80
81 ~PredicateNotifier() {
82 // Ensure there are no waiters.
83 assert((state_.load() & (kStackMask | kWaiterMask)) == kStackMask);
84 }

90 void PrepareWait(Waiter* w) {
91 w->epoch = state_.fetch_add(kWaiterInc, std::memory_order_relaxed);
92 std::atomic_thread_fence(std::memory_order_seq_cst);
93 }
94
97 void CommitWait(Waiter* w) {
98 w->state = Waiter::kNotSignaled;
99 // Modification epoch of this waiter.
100 uint64_t epoch = (w->epoch & kEpochMask) + (((w->epoch & kWaiterMask) >> kWaiterShift) << kEpochShift);
101 uint64_t state = state_.load(std::memory_order_seq_cst);
102 for (;;) {
103 if (int64_t((state & kEpochMask) - epoch) < 0) {
104 // The preceeding waiter has not decided on its fate. Wait until it
105 // calls either CancelWait or CommitWait, or is notified.
106 std::this_thread::yield();
107 state = state_.load(std::memory_order_seq_cst);
108 continue;
109 }
110 // We've already been notified.
111 if (int64_t((state & kEpochMask) - epoch) > 0) return;
112 // Remove this thread from prewait counter and add it to the waiter list.
113 assert((state & kWaiterMask) != 0);
114 uint64_t newstate = state - kWaiterInc + kEpochInc;
115 // newstate = (newstate & ~kStackMask) | (w - &waiters_[0]);
116 newstate = static_cast<uint64_t>((newstate & ~kStackMask) | static_cast<uint64_t>(w - &waiters_[0]));
117 if ((state & kStackMask) == kStackMask)
118 w->next.store(nullptr, std::memory_order_relaxed);
119 else
120 w->next.store(&waiters_[state & kStackMask], std::memory_order_relaxed);
121 if (state_.compare_exchange_weak(state, newstate, std::memory_order_release)) break;
122 }
123 Park(w);
124 }
125
128 void CancelWait(Waiter* w) {
129 uint64_t epoch = (w->epoch & kEpochMask) + (((w->epoch & kWaiterMask) >> kWaiterShift) << kEpochShift);
130 uint64_t state = state_.load(std::memory_order_relaxed);
131 for (;;) {
132 if (int64_t((state & kEpochMask) - epoch) < 0) {
133 // The preceeding waiter has not decided on its fate. Wait until it
134 // calls either CancelWait or CommitWait, or is notified.
135 std::this_thread::yield();
136 state = state_.load(std::memory_order_relaxed);
137 continue;
138 }
139 // We've already been notified.
140 if (int64_t((state & kEpochMask) - epoch) > 0) return;
141 // Remove this thread from prewait counter.
142 assert((state & kWaiterMask) != 0);
143 if (state_.compare_exchange_weak(state, state - kWaiterInc + kEpochInc, std::memory_order_relaxed)) return;
144 }
145 }
146
150 void Notify(bool all) {
151 std::atomic_thread_fence(std::memory_order_seq_cst);
152 uint64_t state = state_.load(std::memory_order_acquire);
153 for (;;) {
154 // Easy case: no waiters.
155 if ((state & kStackMask) == kStackMask && (state & kWaiterMask) == 0) return;
156 uint64_t waiters = (state & kWaiterMask) >> kWaiterShift;
157 uint64_t newstate;
158 if (all) {
159 // Reset prewait counter and empty wait list.
160 newstate = (state & kEpochMask) + (kEpochInc * waiters) + kStackMask;
161 } else if (waiters) {
162 // There is a thread in pre-wait state, unblock it.
163 newstate = state + kEpochInc - kWaiterInc;
164 } else {
165 // Pop a waiter from list and unpark it.
166 Waiter* w = &waiters_[state & kStackMask];
167 Waiter* wnext = w->next.load(std::memory_order_relaxed);
168 uint64_t next = kStackMask;
169 // if (wnext != nullptr) next = wnext - &waiters_[0];
170 if (wnext != nullptr) next = static_cast<uint64_t>(wnext - &waiters_[0]);
171 // Note: we don't add kEpochInc here. ABA problem on the lock-free stack
172 // can't happen because a waiter is re-pushed onto the stack only after
173 // it was in the pre-wait state which inevitably leads to epoch
174 // increment.
175 newstate = (state & kEpochMask) + next;
176 }
177 if (state_.compare_exchange_weak(state, newstate, std::memory_order_acquire)) {
178 if (!all && waiters) return; // unblocked pre-wait thread
179 if ((state & kStackMask) == kStackMask) return;
180 Waiter* w = &waiters_[state & kStackMask];
181 if (!all) w->next.store(nullptr, std::memory_order_relaxed);
182 Unpark(w);
183 return;
184 }
185 }
186 }

190 void NotifyN(size_t n) {
191 if (n >= waiters_.size()) {
192 Notify(true);
193 } else {
194 for (size_t k = 0; k < n; ++k) {
195 Notify(false);
196 }
197 }
198 }

206 Waiter& GetWaiter(size_t idx) { return waiters_[idx]; }

208 private:
209 // State_ layout:
210 // - low kStackBits is a stack of waiters committed wait.
211 // - next kWaiterBits is count of waiters in prewait state.
212 // - next kEpochBits is modification counter.
213 static const uint64_t kStackBits = 16;
214 static const uint64_t kStackMask = (1ull << kStackBits) - 1;
215 static const uint64_t kWaiterBits = 16;
216 static const uint64_t kWaiterShift = 16;
217 static const uint64_t kWaiterMask = ((1ull << kWaiterBits) - 1) << kWaiterShift;
218 static const uint64_t kWaiterInc = 1ull << kWaiterBits;
219 static const uint64_t kEpochBits = 32;
220 static const uint64_t kEpochShift = 32;
221 static const uint64_t kEpochMask = ((1ull << kEpochBits) - 1) << kEpochShift;
222 static const uint64_t kEpochInc = 1ull << kEpochShift;
223 std::atomic<uint64_t> state_;
224 std::vector<Waiter> waiters_;
225
226 void Park(Waiter* w) {
227 std::unique_lock<std::mutex> lock(w->mu);
228 while (w->state != Waiter::kSignaled) {
229 w->state = Waiter::kWaiting;
230 w->cv.wait(lock);
231 }
232 }
233
234 void Unpark(Waiter* waiters) {
235 Waiter* next = nullptr;
236 for (Waiter* w = waiters; w; w = next) {
237 next = w->next.load(std::memory_order_relaxed);
238 unsigned state;
239 {
240 std::unique_lock<std::mutex> lock(w->mu);
241 state = w->state;
242 w->state = Waiter::kSignaled;
243 }
244 // Avoid notifying if it wasn't waiting.
245 if (state == Waiter::kWaiting) w->cv.notify_one();
246 }
247 }
