67template <typename T>
68struct Futurize : public FuturizeBase<T> {
69 using Base = FuturizeBase<T>;
70 using Type = typename Base::Type;
71 using Base::Convert;

73 template <typename Func, typename... FuncArgs>
74 static inline Type Invoke(Func&& func, FuncArgs&&... args) noexcept {
75 try {
76 using ret_t = decltype(func(std::forward<FuncArgs>(args)...));
77 if constexpr (std::is_void_v<ret_t>) {
78 // When func returns void.
79 func(std::forward<FuncArgs>(args)...);
80 // Return Future<>
81 return MakeReadyFuture<>();
82 } else if constexpr (IsFuture<ret_t>::value) {
83 // When func returns Future<T>.
84 return func(std::forward<FuncArgs>(args)...);
85 } else {
86 // When func returns T, then return Future<T>.
87 return Convert(func(std::forward<FuncArgs>(args)...));
88 }
89 } catch (...) {
90 // Exception thrown in func, then return Future with exception.
91 return Type(LockException("WithLock invoke err", LockException::LOCK_EXCEPT));
92 }
93 }
94};

96template<typename Func, typename... Args>
97auto FuturizeInvoke(Func&& func, Args&&... args) noexcept {
98 return Futurize<std::invoke_result_t<Func, Args&&...>>::Invoke(
99 std::forward<Func>(func), std::forward<Args>(args)...);
100}

106// WithLock will wrap it up, e.g. int -> Future<int>.
107template<typename Lock, typename Func>
108inline auto WithLock(Lock* lock, Func&& func) {
109 return lock->Lock().Then([lock, func = std::forward<Func>(func)] () mutable {
110 auto ret = FuturizeInvoke(func);
111 using futurized_type = decltype(ret);
112 return ret.Then([lock] (futurized_type&& ret) {
113 lock->UnLock();
114 return std::move(ret);
115 });
116 });
