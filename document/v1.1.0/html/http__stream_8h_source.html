58 template <typename Clock, typename Dur>
59 Status Read(NoncontiguousBuffer& item, size_t max_bytes, const std::chrono::time_point<Clock, Dur>& expiry) {
60 if (state_ & kErrorBit) {
61 return kStreamStatusServerNetworkError;
62 } else if (state_ & kEofBit) {
63 return kStreamStatusReadEof;
64 }
65
66 if (!read_buffer_.Cut(item, max_bytes, expiry)) {
67 return kStreamStatusServerReadTimeout;
68 }
69
70 if (state_ & kErrorBit) { // encounter network error during the Cut operation
71 return kStreamStatusServerNetworkError;
72 }
73
74 // mark EOF if the data is less than max_bytes
75 // The first encounter with EOF during reading is considered a normal state, while subsequent attempts to read will
76 // return kStreamStatusReadEof.
77 if (item.ByteSize() < max_bytes) {
78 state_ |= kEofBit;
79 }
80
81 return kSuccStatus;
82 }
