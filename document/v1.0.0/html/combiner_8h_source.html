63template <typename T>
64class ElementContainer {
65 public:
67 void Load(T* out) {
68 std::lock_guard<std::mutex> guard(mutex_);
69 *out = value_;
70 }
71
73 void Store(const T& new_value) {
74 std::lock_guard<std::mutex> guard(mutex_);
75 value_ = new_value;
76 }
77
79 void Exchange(T* prev, const T& new_value) {
80 std::lock_guard<std::mutex> guard(mutex_);
81 *prev = value_;
82 value_ = new_value;
83 }
84
86 template <typename Op, typename T1>
87 void Modify(const Op& op, const T1& value2) {
88 std::lock_guard<std::mutex> guard(mutex_);
89 CallOpReturningVoid(op, value_, value2);
90 }
91
92 private:
93 // Raw value to make it thread safe.
94 T value_;
95 // Lock to protect the value.
96 mutable std::mutex mutex_;
97};

101template <typename ResultType, typename ElementType, typename BinaryOp>
102class BlockCombiner {
103 public:
104 using SelfType = BlockCombiner<ResultType, ElementType, BinaryOp>;
105
107 struct Block : public LinkNode<Block> {
109 Block() : combiner_(nullptr) {}
110
112 ~Block() {
113 if (combiner_) {
114 combiner_->CommitAndErase(this);
115 combiner_ = nullptr;
116 }
117 }
118
120 void Reset(const ElementType& val, SelfType* self_type_ptr) {
121 combiner_ = self_type_ptr;
122 element_.Store(val);
123 }
124
125 // Upper combiner current block belongs to.
126 SelfType* combiner_;
127 // Current block value.
128 ElementContainer<ElementType> element_;
129 };
130
131 using RpczBlockGroup = BlockGroup<Block>;
132
134 explicit BlockCombiner(const ResultType& result_identity = ResultType(),
135 const ElementType& element_identity = ElementType(),
136 const BinaryOp& op = BinaryOp())
137 : id_(0), op_(op), global_result_(result_identity), element_identity_(element_identity) {}
138
140 ~BlockCombiner() {
141 if (id_ >= 0) {
142 ClearAllBlock();
143 id_ = -1;
144 }
145 }
146
148 ResultType ResetAllBlocks() {
149 ElementType prev;
150 std::lock_guard<std::mutex> lock(mutex_);
151 ResultType reset_result = nullptr;
152 for (LinkNode<Block>* node = blocks_.head(); node != blocks_.end(); node = node->next()) {
153 node->value()->element_.Exchange(&prev, element_identity_);
154 CallOpReturningVoid(op_, reset_result, prev);
155 }
156 return reset_result;
157 }
158
160 void CommitAndErase(Block* block) {
161 if (block == nullptr) {
162 return;
163 }
164 ElementType local;
165 std::lock_guard<std::mutex> lock(mutex_);
166 block->element_.Load(&local);
167 CallOpReturningVoid(op_, global_result_, local);
168 block->RemoveFromList();
169 }
170
172 Block* GetOrCreateTlsBlock() {
173 Block* block = RpczBlockGroup::GetTlsBlock(id_);
174 if (!block) {
175 block = RpczBlockGroup::GetOrCreateTlsBlock(id_);
176 if (nullptr == block) {
177 TRPC_FMT_ERROR("Fail to create block");
178 return nullptr;
179 }
180 }
181
182 if (block->combiner_) {
183 return block;
184 }
185
186 block->Reset(element_identity_, this);
187
188 {
189 std::lock_guard<std::mutex> lock(mutex_);
190 blocks_.Append(block);
191 }
192 return block;
193 }
194
196 void ClearAllBlock() {
197 std::lock_guard<std::mutex> lock(mutex_);
198 for (LinkNode<Block>* node = blocks_.head(); node != blocks_.end();) {
199 node->value()->Reset(ElementType(), nullptr);
200 LinkNode<Block>* const saved_next = node->next();
201 node->RemoveFromList();
202 node = saved_next;
203 }
204 }
205
207 const BinaryOp& GetOp() const { return op_; }
208
209 private:
210 // block id.
211 uint16_t id_;
212 // Operation to merge thread local block.
213 BinaryOp op_;
214 // To protect global blocks.
215 std::mutex mutex_;
216 // Merged result.
217 ResultType global_result_;
218 // Initial value for each block.
219 ElementType element_identity_;
220 // Global blocks.
221 LinkedList<Block> blocks_;
222};
