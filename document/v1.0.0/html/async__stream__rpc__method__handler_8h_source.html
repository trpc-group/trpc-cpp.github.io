30template <class RequestType, class ResponseType>
31class AsyncStreamRpcMethodHandler final : public RpcMethodHandlerInterface {
32 using R = RequestType;
33 using W = ResponseType;
34
35 public:
36 using ServerStreamFunc = std::function<Future<>(const ServerContextPtr&, R&&, const stream::AsyncWriterPtr<W>&)>;
37
38 using ClientStreamFunc = std::function<Future<W>(const ServerContextPtr&, const stream::AsyncReaderPtr<R>&)>;
39
40 using BidiStreamFunc = std::function<Future<>(const ServerContextPtr&, const stream::AsyncReaderWriterPtr<R, W>&)>;
41
42 explicit AsyncStreamRpcMethodHandler(const ServerStreamFunc& func)
43 : server_stream_func_(func), type_(MethodType::SERVER_STREAMING) {}
44
45 explicit AsyncStreamRpcMethodHandler(const ClientStreamFunc& func)
46 : client_stream_func_(func), type_(MethodType::CLIENT_STREAMING) {}
47
48 explicit AsyncStreamRpcMethodHandler(const BidiStreamFunc& func)
49 : bidi_stream_func_(func), type_(MethodType::BIDI_STREAMING) {}
50
51 void Execute(const ServerContextPtr& context) noexcept override {
52 auto provider = context->GetStreamReaderWriterProvider();
53
54 provider->AsyncStart()
55 .Then([this, context, provider]() {
56 stream::MessageContentCodecOptions content_codec_ops{
57 .serialization = serialization::SerializationFactory::GetInstance()->Get(context->GetReqEncodeType()),
58 .content_type = context->GetReqEncodeType(),
59 };
60
61 auto rw = MakeRefCounted<stream::AsyncReaderWriter<R, W>>(provider, content_codec_ops);
62
63 switch (type_) {
64 case MethodType::SERVER_STREAMING:
65 return rw->Read().Then([this, context, rw](std::optional<R>&& req) {
66 if (!req) {
67 return MakeExceptionFuture<>(stream::StreamError(Status(-1, -1, "unexcepted stream eof")));
68 }
69 return server_stream_func_(context, std::move(req.value()), rw->Writer());
70 });
71 case MethodType::CLIENT_STREAMING:
72 return client_stream_func_(context, rw->Reader()).Then([rw](W&& rsp) {
73 return rw->Write(std::move(rsp));
74 });
75 case MethodType::BIDI_STREAMING:
76 return bidi_stream_func_(context, rw);
77 default:
78 return MakeExceptionFuture<>(stream::StreamError(Status(-1, -1, "unsupported method type")));
79 }
80 })
81 .Then([this, provider](Future<>&& ft) {
82 if (ft.IsFailed()) {
83 auto ex = ft.GetException();
84 // If there is a network error, there is no need to send a reset, because the connection is already unusable
85 // at this point.
86 if (ex.GetExceptionCode() != TRPC_STREAM_SERVER_NETWORK_ERR) {
87 TRPC_FMT_ERROR("Reset server stream with error: {}", ex.what());
88 provider->Reset(ExceptionToStatus(std::move(ex)));
89 }
90 } else {
91 TRPC_FMT_DEBUG("Close server stream.");
92 provider->Close(kSuccStatus);
93 }
94 return MakeReadyFuture<>();
95 }); // future discarded
96 }
97
98 private:
99 Status ExceptionToStatus(Exception&& ex) {
100 if (ex.is<stream::StreamError>()) {
101 return ex.Get<stream::StreamError>()->GetStatus();
102 } else {
103 return Status(ex.GetExceptionCode(), 0, ex.what());
104 }
105 }
106
107 void Execute(const ServerContextPtr& context, NoncontiguousBuffer&& req_data,
108 NoncontiguousBuffer& rsp_data) noexcept override {
109 TRPC_ASSERT(false && "Unreachable");
110 }
111
112 private:
113 ServerStreamFunc server_stream_func_;
114
115 ClientStreamFunc client_stream_func_;
116
117 BidiStreamFunc bidi_stream_func_;
118
119 MethodType type_;
120};
