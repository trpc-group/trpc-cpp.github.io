43template <typename T>
44class Status {
45 public:
47 using WriteBuffer = AtomicType<T>;
49 using SeriesSamplerType = SeriesSampler<Status, T, OpAdd<T>>;
50
51 public:
53 Status() = default;
54
56 explicit Status(T const& value) { value_.Store(value); }
57
61 explicit Status(std::string_view rel_path, T const& value) : Status(value) {
62 handle_ = LinkToParent(rel_path, TrpcVarGroup::FindOrCreate("/"));
63 }
64
69 explicit Status(TrpcVarGroup* parent, std::string_view rel_path, T const& value) : Status(value) {
70 handle_ = LinkToParent(rel_path, parent);
71 }
72
74 ~Status() {
75 if (series_sampler_) {
76 series_sampler_->Destroy();
77 }
78 }
79
82 void Update(T const& value) noexcept { value_.Store(value); }
83
85 void SetValue(T const& value) noexcept { return Update(value); }
86
89 T GetValue() const noexcept { return value_.Load(); }
90
92 std::string GetAbsPath() const { return IsExposed() ? handle_->abs_path : std::string(); }
93
95 bool IsExposed() const { return handle_.has_value(); }
96
99 std::weak_ptr<SeriesSamplerType> GetSeriesSampler() { return series_sampler_; }
100
103 std::unordered_map<std::string, std::vector<T>> GetSeriesValue() const noexcept {
104 if (series_sampler_) {
105 return series_sampler_->GetSeries();
106 }
107 return std::unordered_map<std::string, std::vector<T>>();
108 }
109
110 private:
112 bool CheckSeries() {
113 if constexpr (!std::is_integral_v<T> && !std::is_floating_point_v<T>) {
114 return false;
115 }
116
117 if (!trpc::TrpcConfig::GetInstance()->GetGlobalConfig().tvar_config.save_series) {
118 return false;
119 }
120
121 if (!series_sampler_) {
122 series_sampler_ = std::make_shared<SeriesSamplerType>(this);
123 series_sampler_->Schedule();
124 }
125 return true;
126 }
127
129 std::optional<TrpcVarGroup::Handle> LinkToParent(std::string_view rel_path,
130 TrpcVarGroup* parent) {
131 if (CheckSeries()) {
132 return TrpcVarGroup::LinkToParent(
133 rel_path, parent, [this] { return detail::ToJsonValue(GetValue()); },
134 [this] { return detail::ToJsonValue(GetSeriesValue()); });
135 } else {
136 return TrpcVarGroup::LinkToParent(rel_path, parent,
137 [this] { return detail::ToJsonValue(GetValue()); });
138 }
139 }
140
141 private:
142 // Protected by AtomicType.
143 WriteBuffer value_;
144 // To implement history data.
145 std::shared_ptr<SeriesSamplerType> series_sampler_;
146 // Must be last member.
147 std::optional<TrpcVarGroup::Handle> handle_;
148};
