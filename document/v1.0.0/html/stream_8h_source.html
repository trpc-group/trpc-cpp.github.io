38class StreamReaderImpl : public RefCounted<StreamReaderImpl<R>> {
39 public:

51 Status Read(R* msg, int timeout = -1) {
52 NoncontiguousBuffer msg_buff;
53 auto status = provider_->Read(&msg_buff, timeout);
54 if (!status.OK()) {
55 return status;
56 }
57 if (!Deserialize(&msg_buff, static_cast<void*>(msg))) {
58 status.SetFrameworkRetCode(provider_->GetDecodeErrorCode());
59 status.SetErrorMessage("deserialize message failed");
60 return status;
61 }
62 return status;
63 }
64
66 Status Finish() { return provider_->Finish(); }
67
70 Status GetStatus() const { return provider_->GetStatus(); }
71
72 private:
73 inline bool Deserialize(NoncontiguousBuffer* buffer, void* msg) {
74 serialization::DataType type;
75 if constexpr (std::is_convertible_v<R*, google::protobuf::Message*>) {
76 type = serialization::kPbMessage;
77 } else if constexpr (std::is_convertible_v<R*, rapidjson::Document*>) {
78 type = serialization::kRapidJson;
79 } else if constexpr (std::is_convertible_v<R*, flatbuffers::trpc::MessageFbs*>) {
80 type = serialization::kFlatBuffers;
81 } else if constexpr (std::is_convertible_v<R*, NoncontiguousBuffer*>) {
82 type = serialization::kNonContiguousBufferNoop;
83 } else if constexpr (std::is_convertible_v<R*, std::string*>) {
84 type = serialization::kStringNoop;
85 } else {
86 TRPC_FMT_ERROR("unsupported serialization type");
87 return false;
88 }
89 return content_codec_.serialization->Deserialize(buffer, type, msg);
90 }

106
111 Status Write(const W& msg) {
112 NoncontiguousBuffer msg_buff;
113 if (!Serialize(const_cast<void*>(static_cast<const void*>(&msg)), &msg_buff)) {
114 Status status;
115 status.SetFrameworkRetCode(provider_->GetEncodeErrorCode());
116 status.SetErrorMessage("serialize message failed");
117 return status;
118 }
119 return provider_->Write(std::move(msg_buff));
120 }

332template <typename W, typename R>
333StreamReaderWriter<W, R> Create(const StreamReaderWriterProviderPtr& provider,
334 const MessageContentCodecOptions& content_codec) {
335 TRPC_ASSERT(provider != nullptr && "stream reader writer provider is nullptr");
336 TRPC_ASSERT(content_codec.serialization != nullptr && "serialization is nullptr");
337
338 auto reader_impl = MakeRefCounted<StreamReaderImpl<R>>(provider, content_codec);
339 auto writer_impl = MakeRefCounted<StreamWriterImpl<W>>(provider, content_codec);
340
341 return StreamReaderWriter<W, R>(reader_impl, writer_impl);
342}
