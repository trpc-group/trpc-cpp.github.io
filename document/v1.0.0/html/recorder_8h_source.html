43template <class T>
44struct UpdateAvgBuffer {
45 void operator()(AvgBuffer<T>* left, add_cr_non_integral_t<T> right) const {
46 left->val += right;
47 ++left->num;
48 }
49};

53template <class T>
54struct MergeAvgBuffer {
55 void operator()(AvgBuffer<T>* left, add_cr_non_integral_t<AvgBuffer<T>> right) const {
56 left->val += right.val;
57 left->num += right.num;
58 }
59};

63template <class T>
64struct MinusAvgBuffer {
65 void operator()(AvgBuffer<T>* left, add_cr_non_integral_t<AvgBuffer<T>> right) const {
66 left->val -= right.val;
67 left->num -= right.num;
68 }
69};

78struct CompressedIntAvgTraits {
79 using Type = std::int64_t;
80 using WriteBuffer = AtomicType<Type>;
81 using InputDataType = std::int64_t;
82 using ResultType = AvgBuffer<Type>;
83 using SamplerOp = MergeAvgBuffer<Type>;
84 using SamplerInvOp = MinusAvgBuffer<Type>;
85
86 struct CompressedIntAvg {
87 // Compressing format:
88 // | 20 bits (unsigned) | sign bit | 43 bits |
89 // num sum
90 static const size_t SUM_BIT_WIDTH = 44;
91 static const uint64_t MAX_SUM_PER_THREAD = (1ul << SUM_BIT_WIDTH) - 1;
92 static const uint64_t MAX_NUM_PER_THREAD = (1ul << (64ul - SUM_BIT_WIDTH)) - 1;
93 static_assert(SUM_BIT_WIDTH > 32, "SUM_BIT_WIDTH_must_be_between_33_and_63");
94 static_assert(SUM_BIT_WIDTH < 64, "SUM_BIT_WIDTH_must_be_between_33_and_63");
95
96 static uint64_t GetSum(int64_t n) { return (static_cast<uint64_t>(n) & MAX_SUM_PER_THREAD); }
97
98 static uint64_t GetNum(int64_t n) { return static_cast<uint64_t>(n) >> SUM_BIT_WIDTH; }
99
100 static uint64_t Compress(uint64_t num, uint64_t sum) {
101 return (num << SUM_BIT_WIDTH)
102 // There is a redundant '1' in the front of sum which was
103 // combined with two negative number, so truncation has to be
104 // done here
105 | (sum & MAX_SUM_PER_THREAD);
106 }
107
108 static bool WillOverflow(int64_t lhs, int rhs) {
109 return
110 // Both integers are positive and the sum is larger than the largest
111 // number
112 ((lhs > 0) && (rhs > 0) && (lhs + rhs > ((int64_t)MAX_SUM_PER_THREAD >> 1)))
113 // Or both integers are negative and the sum is less than the lowest
114 // number
115 || ((lhs < 0) && (rhs < 0) && (lhs + rhs < (-((int64_t)MAX_SUM_PER_THREAD >> 1)) - 1));
116 // otherwise the sum cannot overflow iff lhs does not overflow
117 // because |sum| < |lhs|
118 }
119
123 static int64_t ExtendSignBit(uint64_t sum) {
124 return (((1ul << (64ul - SUM_BIT_WIDTH + 1)) - 1) * ((1ul << (SUM_BIT_WIDTH - 1) & sum))) |
125 (int64_t)sum;
126 }
127
129 static uint64_t GetComplement(int64_t n) { return n & (MAX_SUM_PER_THREAD); }
130 };

132 struct MergeIntBuffer {
133 void operator()(ResultType* left, add_cr_non_integral_t<Type> right) {
134 left->val +=
135 CompressedIntAvg::ExtendSignBit(CompressedIntAvg::GetSum(static_cast<uint64_t>(right)));
136 left->num += CompressedIntAvg::GetNum(static_cast<uint64_t>(right));
137 }
138 };

141 template <typename TLS>
142 static void Update(TLS* wb, InputDataType val) {
143 if (TRPC_UNLIKELY(static_cast<int64_t>(static_cast<int>(val)) != val)) {
144 char const* reason = nullptr;
145 if (val > std::numeric_limits<int>::max()) {
146 reason = "over_flows";
147 val = std::numeric_limits<int>::max();
148 } else {
149 reason = "under_flows";
150 val = std::numeric_limits<int>::min();
151 }
152 TRPC_FMT_WARN("Input={} to IntRecorderTraits {}", val, reason);
153 }
154
155 int64_t n = wb->buffer_.Load();
156 uint64_t const complement = CompressedIntAvg::GetComplement(val);
157 uint64_t num = 0;
158 uint64_t sum = 0;
159 do {
160 num = CompressedIntAvg::GetNum(n);
161 sum = CompressedIntAvg::GetSum(n);
162 if (TRPC_UNLIKELY(
163 (num + 1 > CompressedIntAvg::MAX_NUM_PER_THREAD ||
164 CompressedIntAvg::WillOverflow(CompressedIntAvg::ExtendSignBit(sum), val)))) {
165 // Although agent->element might have been cleared at this
166 // point, it is just OK because the very value is 0 in
167 // this case
168 wb->CommitAndClear();
169 sum = 0;
170 num = 0;
171 n = 0;
172 }
173 } while (!wb->buffer_.CompareExchangeWeak(
174 &n, CompressedIntAvg::Compress(num + 1, sum + complement)));
175 }
176
178 static void Merge(ResultType* wb1, add_cr_non_integral_t<Type> wb2) {
179 MergeIntBuffer()(wb1, wb2);
180 }
181};
