26template <class F, class... Args,
27 class R = trpc::futurize_t<std::invoke_result_t<F&&, Args&&...>>>
28R Async(Launch policy, std::size_t scheduling_group,
29 FiberExecutionContext* execution_context, F&& f, Args&&... args) {
30 TRPC_CHECK(policy == Launch::Post || policy == Launch::Dispatch);
31 trpc::as_promise_t<R> p;
32 auto rc = p.GetFuture();
33 auto proc = [p = std::move(p), f = std::forward<F>(f),
34 args = std::make_tuple(std::forward<Args>(args)...)]() mutable {
35 if constexpr (std::is_same_v<Future<>, R>) {
36 std::apply(f, std::move(args));
37 p.SetValue();
38 } else {
39 p.SetValue(std::apply(f, std::move(args)));
40 }
41 };
42
43 trpc::StartFiberDetached(
44 Fiber::Attributes{.launch_policy = policy,
45 .scheduling_group = scheduling_group,
46 .execution_context = execution_context},
47 std::move(proc));
48 return rc;
49}
