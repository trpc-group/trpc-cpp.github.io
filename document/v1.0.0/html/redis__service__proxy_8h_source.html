127Status RedisServiceProxy::UnaryInvoke(const ClientContextPtr& context, RequestMessage&& req, ResponseMessage* rsp) {
128 if (!context->GetRequest()) {
129 context->SetRequest(codec_->CreateRequestPtr());
130 }
131
132 FillClientContext(context);
133
134 context->SetRequestData(&req);
135 context->SetReqEncodeDataType(serialization::kRedisType);
136
137 context->SetRspEncodeType(serialization::kNoopType);
138 context->SetRspEncodeDataType(serialization::kNoopType);
139
140 auto filter_status = filter_controller_.RunMessageClientFilters(FilterPoint::CLIENT_PRE_RPC_INVOKE, context);
141
142 if (filter_status != FilterStatus::REJECT) {
143 UnaryInvokeImp<RequestMessage, ResponseMessage>(context, std::move(req), rsp);
144 } else {
145 TRPC_FMT_ERROR("service name:{}, filter pre execute failed.", GetServiceName());
146 }
147
148 context->SetResponseData(rsp);
149
150 filter_controller_.RunMessageClientFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
151
152 context->SetRequestData(nullptr);
153 context->SetResponseData(nullptr);
154
155 return context->GetStatus();
156}

224template <class RequestMessage, class ResponseMessage>
225Future<ResponseMessage> RedisServiceProxy::AsyncUnaryInvokeImp(const ClientContextPtr& context, RequestMessage&& req) {
226 ProtocolPtr& req_protocol = context->GetRequest();
227
228 if (!codec_->FillRequest(context, req_protocol, reinterpret_cast<void*>(&req))) {
229 std::string error("service name:");
230 error += GetServiceName();
231 error += ", request fill body failed.";
232
233 TRPC_LOG_ERROR(error);
234
235 Status status;
236 status.SetFrameworkRetCode(TrpcRetCode::TRPC_CLIENT_ENCODE_ERR);
237 status.SetErrorMessage(error);
238
239 context->SetStatus(std::move(status));
240
241 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
242
243 return MakeExceptionFuture<ResponseMessage>(CommonException(context->GetStatus().ErrorMessage().c_str()));
244 }
245
246 return ServiceProxy::AsyncUnaryInvoke(context, req_protocol)
247 .Then([this, context](Future<ProtocolPtr>&& rsp_protocol) {
248 if (rsp_protocol.IsFailed()) {
249 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
250 return MakeExceptionFuture<ResponseMessage>(rsp_protocol.GetException());
251 }
252
253 context->SetResponse(rsp_protocol.GetValue0());
254
255 ResponseMessage rsp_obj;
256 void* raw_rsp = static_cast<void*>(&rsp_obj);
257
258 if (!codec_->FillResponse(context, context->GetResponse(), raw_rsp)) {
259 std::string error("service name:");
260 error += GetServiceName();
261 error += ", response fill body failed.";
262
263 TRPC_LOG_ERROR(error);
264
265 Status status;
266 status.SetFrameworkRetCode(TrpcRetCode::TRPC_CLIENT_DECODE_ERR);
267 status.SetErrorMessage(error);
268
269 context->SetStatus(std::move(status));
270
271 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
272 return MakeExceptionFuture<ResponseMessage>(CommonException(context->GetStatus().ErrorMessage().c_str()));
273 }
274
275 context->SetResponseData(&rsp_obj);
276
277 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
278 return MakeReadyFuture<ResponseMessage>(std::move(rsp_obj));
279 });

282template <class RequestMessage>
283Status RedisServiceProxy::OnewayInvoke(const ClientContextPtr& context, RequestMessage&& req) {
284 if (!context->GetRequest()) {
285 context->SetRequest(codec_->CreateRequestPtr());
286 }
287
288 context->SetRspEncodeType(serialization::kNoopType);
289 context->SetRspEncodeDataType(serialization::kNoopType);
290 context->SetCallType(kOnewayCall);
291
292 FillClientContext(context);
293
294 context->SetRequestData(&req);
295 context->SetReqEncodeDataType(serialization::kRedisType);
296
297 auto filter_ret = RunFilters(FilterPoint::CLIENT_PRE_RPC_INVOKE, context);
298 if (filter_ret == 0) {
299 const ProtocolPtr& req_protocol = context->GetRequest();
300
301 if (!codec_->FillRequest(context, req_protocol, reinterpret_cast<void*>(&req))) {
302 std::string error("service name:");
303 error += GetServiceName();
304 error += ", request fill body failed.";
305 TRPC_LOG_ERROR(error);
306
307 Status status;
308 status.SetFrameworkRetCode(TrpcRetCode::TRPC_CLIENT_ENCODE_ERR);
309 status.SetErrorMessage(error);
310
311 context->SetStatus(status);
312
313 return status;
314 }
315
316 ServiceProxy::OnewayInvoke(context, req_protocol);
317 }
318
319 RunFilters(FilterPoint::CLIENT_POST_RPC_INVOKE, context);
320 context->SetRequestData(nullptr);
321 return context->GetStatus();
322}
