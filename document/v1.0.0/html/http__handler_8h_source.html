46 virtual Status Handle(const ServerContextPtr& ctx, const HttpRequestPtr& req, HttpResponse* rsp) {
47 Status status;
48 try {
49 switch (req->GetMethodType()) {
50 TRPC_HTTP_HANDLER_CASE(GET, Get);
51 TRPC_HTTP_HANDLER_CASE(HEAD, Head);
52 TRPC_HTTP_HANDLER_CASE(POST, Post);
53 TRPC_HTTP_HANDLER_CASE(PUT, Put);
54 TRPC_HTTP_HANDLER_CASE(DELETE, Delete);
55 TRPC_HTTP_HANDLER_CASE(OPTIONS, Options);
56 TRPC_HTTP_HANDLER_CASE(PATCH, Patch);
57 default:
58 rsp->SetStatus(ResponseStatus::kNotImplemented);
59 }
60 } catch (const BaseException& http_ex) {
61 rsp->SetStatus(http_ex.Status(), http_ex.str());
62 } catch (...) { // Do not return other exception messages to users for safety reasons.
63 rsp->SetStatus(ResponseStatus::kInternalServerError);
64 }
65 rsp->Done();
66 return status;
67 }
