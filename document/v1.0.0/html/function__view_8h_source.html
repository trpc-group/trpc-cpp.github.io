12// Type-erasing wrapper for `Callable`s.
13//
14// INTENDED FOR USE IN FUNCTION PARAMETER, **ONLY**.
15//
16// `Callable`s are neither required to be `CopyConstructible` nor
17// `MoveConstructible`. They're not copy / moved at all.
18//
19// When applicable, this wrapper is more efficient than `std::function` /
20// `Function`.
21//
22// BE AWARE, `FUNCTOR` USED TO CONSTRUCT `FUNCTIONVIEW` IS EXPECTED TO LIVE
23// THROUGH `FUNCTIONVIEW`'S LIFETIME.
24//
25// Normally this is the case if `FunctionView` is used as function parameter and
26// is not leaked outside of the function. The spec explicitly requires all
27// temporaries created in calling function (including those for constructing
28// `FunctionView`) shall be alive until the function-call expression ends.
29

33#include <type_traits>
34#include <utility>

36namespace trpc {

38template <class T>
39class FunctionView;

45template <class R, class... Args, bool kNoexcept>
46class FunctionView<R(Args...) noexcept(kNoexcept)> {
47 public:
48 // Construct
49 FunctionView() : invoker_(nullptr) {}
50
54 template <class T>
55 static constexpr bool implicitly_convertible_v =
56 std::is_invocable_r_v<R, T, Args...> && !std::is_same_v<std::decay_t<T>, FunctionView> &&
57 (!kNoexcept || std::is_nothrow_invocable_v<T, Args...>);
58
63 template <class T, class = std::enable_if_t<implicitly_convertible_v<T>>>
64 constexpr FunctionView(T&& functor) {
65 using RefRemoved = std::remove_reference_t<T>;
66 constexpr auto kConstantFunctor = std::is_const_v<RefRemoved>;
67
68 // `functor` is casted to `const ...` unconditionally to simplify coding.
69 //
70 // We'll strip the `const` if appropriate in `invoker_`.
71 object_ = reinterpret_cast<const void*>(&functor);
72 invoker_ = [](const void* object, Args&&... args) -> R {
73 if constexpr (kConstantFunctor) {
74 return Invoke(std::forward<T>(*reinterpret_cast<RefRemoved*>(object)), std::forward<Args>(args)...);
75 } else {
76 // Otherwise we need to strip `const` we've added to `object_`.
77 return Invoke(std::forward<T>(*reinterpret_cast<RefRemoved*>(const_cast<void*>(object))),
78 std::forward<Args>(args)...);
79 }
80 };
81 }
82
84 R operator()(Args... args) const noexcept(kNoexcept) { return invoker_(object_, std::forward<Args>(args)...); }
85
87 explicit operator bool() const { return !!invoker_; }
88
89 private:
90 using Invoker = R (*)(const void* object, Args&&... args);
91
92 // Helper method for calling `functor`. Specializes for case where `R` is
93 // `void`.
94 template <class T>
95 static R Invoke(T&& functor, Args&&... args) noexcept(kNoexcept) {
96 if constexpr (std::is_void_v<R>) {
97 std::invoke(std::forward<T>(functor), std::forward<Args>(args)...);
98 } else {
99 return std::invoke(std::forward<T>(functor), std::forward<Args>(args)...);
100 }
101 }
102
103 const void* object_;
104 Invoker invoker_;
105};
