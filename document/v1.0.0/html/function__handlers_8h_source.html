75 explicit FuncHandler(const ServeJsonFunction& json_request_function)
76 : handle_function_([json_request_function](trpc::ServerContextPtr context, trpc::http::RequestPtr req,
77 trpc::http::Response* rsp) {
78 rapidjson::Document content;
79 auto status = json_request_function(std::move(context), std::move(req), content);
80 // Converts JSON document to string.
81 rapidjson::StringBuffer buffer;
82 rapidjson::Writer<rapidjson::StringBuffer> writer(buffer);
83 content.Accept(writer);
84 rsp->GetMutableContent()->append(buffer.GetString());
85 return status;
86 }),
