44template <class Traits>
45class WriteMostly {
46 template <typename>
47 friend class GlobalValue;
48
49 public:
50 using Type = typename Traits::Type;
51 using WriteBuffer = typename Traits::WriteBuffer;
52 using ResultType = typename Traits::ResultType;
53 using InputDataType = typename Traits::InputDataType;
54 using SamplerOp = typename Traits::SamplerOp;
55 using SamplerInvOp = typename Traits::SamplerInvOp;
56 using ReducerSamplerType = ReducerSampler<WriteMostly, ResultType, SamplerOp, SamplerInvOp>;
57 using SeriesSamplerType = SeriesSampler<WriteMostly, ResultType, SamplerOp>;
58
59 public:
60 explicit WriteMostly(ResultType r = ResultType(), Type t = Type())
61 : result_identity_(std::move(r)),
62 buffer_identity_(std::move(t)),
63 exited_thread_combined_(result_identity_),
64 tls_buffer_([this]() { return new WriteBufferWrapper(this); }) {}
65
67 virtual ~WriteMostly() {
68 if (reducer_sampler_) {
69 reducer_sampler_->Destroy();
70 }
71 if (series_sampler_) {
72 series_sampler_->Destroy();
73 }
74 }

77 std::weak_ptr<ReducerSamplerType> GetReducerSampler() {
78 if (!reducer_sampler_) {
79 reducer_sampler_ = std::make_shared<ReducerSamplerType>(this);
80 reducer_sampler_->Schedule();
81 }
82 return reducer_sampler_;
83 }
84
86 std::weak_ptr<SeriesSamplerType> GetSeriesSampler() { return series_sampler_; }
87
89 std::unordered_map<std::string, std::vector<ResultType>> GetSeriesValue() const noexcept {
90 if (series_sampler_) {
91 return series_sampler_->GetSeries();
92 }
93 return std::unordered_map<std::string, std::vector<ResultType>>();
94 }
95
97 void Update(InputDataType value) noexcept { Traits::Update(tls_buffer_.Get(), value); }
98
100 ResultType Reset() noexcept {
101 Type prev;
102 std::unique_lock<std::mutex> lock(mutex_);
103 ResultType ret = exited_thread_combined_;
104 exited_thread_combined_ = result_identity_;
105 tls_buffer_.ForEach([&](auto&& wrapper) {
106 wrapper->buffer_.Exchange(&prev, buffer_identity_);
107 Traits::Merge(&ret, prev);
108 });
109 return ret;
110 }
111
113 ResultType GetValue() const noexcept {
114 std::unique_lock<std::mutex> lock(mutex_);
115 ResultType ret = exited_thread_combined_;
116 tls_buffer_.ForEach([&](auto&& wrapper) { Traits::Merge(&ret, wrapper->buffer_.Load()); });
117 return ret;
118 }

122 struct alignas(hardware_destructive_interference_size) WriteBufferWrapper {
123 using ParentType = WriteMostly;
124
125 explicit WriteBufferWrapper(WriteMostly* parent) : parent_(parent) {
126 buffer_.Store(parent_->buffer_identity_);
127 }
128
130 void CommitAndClear() {
131 Type prev;
132 std::unique_lock<std::mutex> lock(parent_->mutex_);
133 buffer_.Exchange(&prev, parent_->buffer_identity_);
134 Traits::Merge(&parent_->exited_thread_combined_, prev);
135 }
136
138 ~WriteBufferWrapper() {
139 std::unique_lock<std::mutex> lock(parent_->mutex_);
140 Traits::Merge(&parent_->exited_thread_combined_, buffer_.Load());
141 }

144 WriteMostly* parent_;
145 WriteBuffer buffer_;
146 };

150 std::optional<TrpcVarGroup::Handle> LinkToParent(std::string_view rel_path,
151 TrpcVarGroup* parent) {
152 if (CheckSeries()) {
153 return TrpcVarGroup::LinkToParent(
154 rel_path, parent, [this] { return detail::ToJsonValue(GetValue()); },
155 [this] { return detail::ToJsonValue(GetSeriesValue()); });
156 } else {
157 return TrpcVarGroup::LinkToParent(rel_path, parent,
158 [this] { return detail::ToJsonValue(GetValue()); });
159 }
160 }

163 std::optional<TrpcVarGroup::Handle> LinkToParent(std::string_view rel_path) {
164 return LinkToParent(rel_path, TrpcVarGroup::FindOrCreate("/"));
165 }

167 private:
169 bool CheckSeries() {
170 if constexpr (is_void_op<SamplerInvOp>::value ||
171 (!std::is_integral_v<ResultType> && !std::is_floating_point_v<ResultType>)) {
172 return false;
173 }
174
175 if (!trpc::TrpcConfig::GetInstance()->GetGlobalConfig().tvar_config.save_series) {
176 return false;
177 }
178
179 if (!series_sampler_) {
180 series_sampler_ = std::make_shared<SeriesSamplerType>(this);
181 series_sampler_->Schedule();
182 }
183 return true;
184 }
185
186 private:
187 mutable std::mutex mutex_;
188 // Used to init exited_thread_combined_.
189 ResultType result_identity_;
190 // Used to init WriteBufferWrapper::buffer_.
191 Type buffer_identity_;
192 // Store thread local data of thread exiting.
193 mutable ResultType exited_thread_combined_;
194 // Core member, thread local wrapper for every threads.
195 ThreadLocal<WriteBufferWrapper> tls_buffer_;
196 // Used to implement Window/PerSecond.
197 std::shared_ptr<ReducerSamplerType> reducer_sampler_;
198 // Used to implement history data.
199 std::shared_ptr<SeriesSamplerType> series_sampler_;
200};
