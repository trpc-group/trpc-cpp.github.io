31template <typename T>
32class alignas(64) UnboundedSPMCQueue {
33 public:
34 static_assert(std::is_pointer_v<T>, "T must be a pointer type");

61 private:
62 struct alignas(64) Array {
63 int64_t cap;
64 int64_t mod;
65 std::atomic<T>* storage;
66
67 explicit Array(int64_t c) : cap{c}, mod{c - 1}, storage{new std::atomic<T>[static_cast<size_t>(cap)]} {}
68
69 ~Array() { delete[] storage; }
70
71 int64_t Capacity() const noexcept { return cap; }
72
73 void Push(int64_t i, T o) noexcept { storage[i & mod].store(o, std::memory_order_relaxed); }
74
75 T Pop(int64_t i) noexcept { return storage[i & mod].load(std::memory_order_relaxed); }
76
77 Array* Resize(int64_t bottom, int64_t top) {
78 Array* ptr = new Array{2 * cap};
79 for (int64_t i = top; i != bottom; ++i) {
80 ptr->Push(i, Pop(i));
81 }
82 return ptr;
83 }
84 };
85
86 alignas(64) std::atomic<int64_t> top_;
87 alignas(64) std::atomic<int64_t> bottom_;
88 alignas(64) std::atomic<Array*> array_;
89 std::vector<Array*> garbage_;
90};

92template <typename T>
93UnboundedSPMCQueue<T>::UnboundedSPMCQueue(int64_t c) {
94 assert(c && (!(c & (c - 1))));
95 top_.store(0, std::memory_order_relaxed);
96 bottom_.store(0, std::memory_order_relaxed);
97 array_.store(new Array{c}, std::memory_order_relaxed);
98 garbage_.reserve(32);
99}

123template <typename T>
124void UnboundedSPMCQueue<T>::Push(T o) {
125 int64_t b = bottom_.load(std::memory_order_relaxed);
126 int64_t t = top_.load(std::memory_order_acquire);
127 Array* a = array_.load(std::memory_order_relaxed);
128
129 if (a->Capacity() - 1 < (b - t)) {
130 Array* tmp = a->Resize(b, t);
131 garbage_.push_back(a);
132 std::swap(a, tmp);
133 array_.store(a, std::memory_order_release);
134 }
135
136 a->Push(b, o);
137 std::atomic_thread_fence(std::memory_order_release);
138 bottom_.store(b + 1, std::memory_order_relaxed);
139}

141template <typename T>
142T UnboundedSPMCQueue<T>::Pop() {
143 int64_t b = bottom_.load(std::memory_order_relaxed) - 1;
144 Array* a = array_.load(std::memory_order_relaxed);
145 bottom_.store(b, std::memory_order_relaxed);
146 std::atomic_thread_fence(std::memory_order_seq_cst);
147 int64_t t = top_.load(std::memory_order_relaxed);
148
149 T item{nullptr};
150
151 if (t <= b) {
152 item = a->Pop(b);
153 if (t == b) {
154 // the last item just got stolen
155 if (!top_.compare_exchange_strong(t, t + 1, std::memory_order_seq_cst, std::memory_order_relaxed)) {
156 item = nullptr;
157 }
158 bottom_.store(b + 1, std::memory_order_relaxed);
159 }
160 } else {
161 bottom_.store(b + 1, std::memory_order_relaxed);
162 }
163
164 return item;
165}

167template <typename T>
168T UnboundedSPMCQueue<T>::Steal() {
169 int64_t t = top_.load(std::memory_order_acquire);
170 std::atomic_thread_fence(std::memory_order_seq_cst);
171 int64_t b = bottom_.load(std::memory_order_acquire);
172
173 T item{nullptr};
174
175 if (t < b) {
176 Array* a = array_.load(std::memory_order_consume);
177 item = a->Pop(t);
178 if (!top_.compare_exchange_strong(t, t + 1, std::memory_order_seq_cst, std::memory_order_relaxed)) {
179 return nullptr;
180 }
181 }
182
183 return item;
184}
