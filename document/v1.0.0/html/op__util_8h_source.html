111template <typename TLS>
112class GlobalValue {
113 public:
114 using WriteMostlyType = typename TLS::ParentType;
115 using WriteBufferType = typename WriteMostlyType::WriteBuffer;
116 using ResultType = typename WriteMostlyType::ResultType;
117
118 explicit GlobalValue(TLS* wrapper) : write_buffer_(&wrapper->buffer_), write_mostly_(wrapper->parent_) {}
119
120 ResultType* Lock() {
121 write_buffer_->mutex_.unlock();
122 write_mostly_->mutex_.lock();
123 return &(write_mostly_->exited_thread_combined_);
124 }
125
126 void Unlock() {
127 write_mostly_->mutex_.unlock();
128 write_buffer_->mutex_.lock();
129 }
130
131 private:
132 WriteBufferType* write_buffer_;
133 WriteMostlyType* write_mostly_;
134};

140template <class T>
141Json::Value ToJsonValue(const T& t) {
142 if constexpr (std::is_same_v<T, Json::Value>) {
143 return t;
144 } else if constexpr (std::is_integral_v<T>) {
145 if constexpr (std::is_unsigned_v<T>) {
146 return Json::Value(static_cast<Json::UInt64>(t));
147 } else {
148 return Json::Value(static_cast<Json::Int64>(t));
149 }
150 } else if constexpr (std::is_floating_point_v<T> ||
151 std::is_same_v<T, std::string_view> || std::is_same_v<T, std::string> ||
152 std::is_same_v<T, const char*> || std::is_same_v<T, bool>) {
153 return Json::Value(t);
154 } else {
155 return Json::Value(t.ToString());
156 }
157}

161template <class T>
162Json::Value ToJsonValue(const AvgBuffer<T>& v) {
163 return ToJsonValue(v.Average());
164}

168template <class T>
169Json::Value ToJsonValue(const std::vector<T>& v) {
170 Json::Value value;
171 for (auto&& element : v) {
172 value.append(ToJsonValue(element));
173 }
174 return value;
175}

179template <class T>
180Json::Value ToJsonValue(const std::unordered_map<std::string, T>& v) {
181 Json::Value value;
182 for (auto&& kv : v) {
183 value[kv.first] = ToJsonValue(kv.second);
184 }
185 return value;
186}
