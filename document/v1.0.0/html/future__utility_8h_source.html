36template <class... Ts>
37Future<Ts...> BlockingGet(Future<Ts...>&& f) {
38 Latch latch(1);
39 Future<Ts...> future;
40 std::move(f).Then([&](Future<Ts...>&& fut) {
41 future = std::move(fut);
42 latch.count_down();
43 return MakeReadyFuture<>();
44 });
45 latch.wait();
46
47 return future;
48}

55template <class... Ts>
56std::optional<Future<Ts...>> BlockingTryGet(Future<Ts...>&& f, uint32_t timeout_ms) {
57 struct State {
58 Future<Ts...> future;
59 Latch latch{1};
60 };
61
62 auto state = std::make_shared<State>();
63 std::move(f).Then([state](Future<Ts...>&& fut) {
64 state->future = std::move(fut);
65 state->latch.count_down();
66 return MakeReadyFuture<>();
67 });
68
69 bool ret = state->latch.wait_for(std::chrono::milliseconds(timeout_ms));
70 if (!ret) {
71 return std::nullopt;
72 }
73
74 return std::move(state->future);
75}

95template <typename Stop, typename Func, typename R = typename std::invoke_result_t<Func>>
96struct DoUntilState {
97 Stop stop;
98 Func func;
99 Promise<> pr;
100
101 DoUntilState(Stop&& s, Func&& f) : stop(std::forward<Stop>(s)), func(std::forward<Func>(f)) {}
102
104 void Run(std::unique_ptr<DoUntilState>&& sp, bool nostop = false) {
105 if (!nostop && stop()) {
106 pr.SetValue();
107 } else {
108 func().Then([sp = std::move(sp)](R&& ft) mutable {
109 if (ft.IsFailed())
110 sp->pr.SetException(ft.GetException());
111 else
112 sp->Run(std::move(sp));
113 return MakeReadyFuture<>();
114 });
115 }
116 }
117};

131template <typename Tuple>
132class WhenAllFunction {
133 public:
134 WhenAllFunction(std::shared_ptr<Promise<Tuple>> pr, Tuple &futs,
135 const std::shared_ptr<std::atomic_uint32_t> &counter)
136 : pr_(pr), futs_(futs), counter_(counter) {}
137
138 template <typename Fut>
139 void operator()(Fut &fut) {
140 auto &futs = futs_;
141 auto &input_fut = fut;
142 DoWith(std::move(fut), [&futs, &input_fut, counter = counter_, pr = pr_](Fut &fut) mutable {
143 return fut.Then(
144 [&futs, &input_fut, counter, pr](std::remove_reference_t<Fut> &&result) mutable {
145 input_fut = std::move(result);
146 auto current = (*counter).fetch_add(1);
147 if (current + 1 == std::tuple_size<Tuple>::value) {
148 pr->SetValue(std::move(futs));
149 }
150 auto future = MakeReadyFuture<>();
151 return future;
152 });
153 });
154 }
155
156 private:
157 std::shared_ptr<Promise<Tuple>> pr_;
158
159 Tuple &futs_;
160
161 std::shared_ptr<std::atomic_uint32_t> counter_;
162};

180template <typename Stop, typename Func, typename R = std::invoke_result_t<Func>,
181 typename = std::enable_if_t<std::is_same_v<std::invoke_result_t<Stop>, bool>>,
182 typename = std::enable_if_t<is_future_v<R>>>
183Future<> DoUntil(Stop&& stop, Func&& func) {
184 if (stop()) {
185 return MakeReadyFuture<>();
186 }
187
188 auto sp = std::make_unique<DoUntilState<Stop, Func>>(std::forward<Stop>(stop), std::forward<Func>(func));
189 auto ft = sp->pr.GetFuture();
190 sp->Run(std::move(sp), true);
191
192 return ft;

235template <typename Iterator, typename Func>
236Future<> DoForEach(Iterator&& begin, Iterator&& end, Func&& func) {
237 if constexpr (std::is_invocable_v<Func, typename Iterator::value_type>) {
238 auto state = std::make_shared<Iterator>(std::forward<Iterator>(begin));
239 return DoUntil([state, end = std::forward<Iterator>(end)]() { return (*state) == end; },
240 [state, func = std::forward<Func>(func)]() { return func(*(*state)++); });
241 } else {
242 static_assert(std::is_invocable_v<Func>);
243 return DoUntil([begin = std::forward<Iterator>(begin),
244 end = std::forward<Iterator>(end)]() mutable {
245 return begin++ == end;
246 },
247 std::forward<Func>(func));
248 }
249}

265template <typename Func>
266Future<> DoFor(std::size_t n, Func&& func) {
267 if constexpr (std::is_invocable_v<Func, std::size_t>) {
268 auto state = std::make_shared<std::size_t>(n);
269 return DoUntil([state]() { return *state == 0; },
270 [n, state, func = std::forward<Func>(func)]() mutable { return func(n - (*state)--); });
271 } else {
272 static_assert(std::is_invocable_v<Func>);
273 return DoUntil([n]() mutable { return n-- == 0; }, std::forward<Func>(func));
274 }

281template <typename... FutsType>
282Future<std::tuple<FutsType...>> WhenAll(FutsType &&... futs) {
283 if constexpr (std::tuple_size<std::tuple<FutsType...>>::value == 0) {
284 return MakeReadyFuture<std::tuple<>>(std::tuple<>());
285 } else {
286 auto input = std::make_tuple(std::move(futs)...);
287
288 return DoWith(std::move(input), [](std::tuple<FutsType...> &futs) {
289 auto promise_ptr = std::make_shared<Promise<std::tuple<FutsType...>>>();
290 auto future = promise_ptr->GetFuture();
291 auto counter = std::make_shared<std::atomic_uint32_t>(0);
292 for_each(
293 futs, WhenAllFunction<std::tuple<FutsType...>>(promise_ptr, futs, counter));
294 return future;
295 });
296 }

304template <typename InputIterator, typename FutureType = typename std::iterator_traits<InputIterator>::value_type>
305Future<std::vector<FutureType>> WhenAll(InputIterator first, InputIterator last) {
306 std::vector<FutureType> result;
307 if (last - first == 0) {
308 return MakeReadyFuture<std::vector<FutureType>>(std::move(result));
309 } else {
310 return DoWith(std::move(result), [last, first](std::vector<FutureType> &result) mutable {
311 result.resize(last - first);
312 auto promise_ptr = std::make_shared<Promise<std::vector<FutureType>>>();
313 auto future = promise_ptr->GetFuture();
314 auto counter = std::make_shared<std::atomic_uint32_t>(0);
315 uint32_t size = last - first;
316 for (auto it = first; it != last; ++it) {
317 uint32_t idx = it - first;
318 it->Then([counter, idx, &result, size, promise = promise_ptr](FutureType &&fut) mutable {
319 result[idx] = std::move(fut);
320 std::atomic_thread_fence(std::memory_order_acq_rel);
321 auto current = (*counter).fetch_add(1);
322 if (current + 1 == size) {
323 promise->SetValue(std::move(result));
324 }
325 return MakeReadyFuture<>();
326 });
327 }
328 return future;
329 });
330 }

338template <typename InputIterator,
339 typename FutureType = typename std::iterator_traits<InputIterator>::value_type,
340 typename FutureValueType = typename FutureType::TupleValueType>
341Future<size_t, FutureValueType> WhenAny(InputIterator first, InputIterator last) {
342 struct InnerContext {
343 Promise<size_t, FutureValueType> promise;
344 std::atomic_bool done{false};
345 };
346
347 if (last - first == 0) {
348 FutureValueType result;
349 return MakeReadyFuture<size_t, FutureValueType>(0, std::move(result));
350 } else {
351 auto context = std::make_shared<InnerContext>();
352 auto future = context->promise.GetFuture();
353 for (auto it = first; it != last; ++it) {
354 uint32_t idx = it - first;
355 it->Then([context, idx](FutureType &&fut) mutable {
356 auto has_done = context->done.exchange(true);
357 if (!has_done) {
358 if (fut.IsReady()) {
359 context->promise.SetValue(idx, std::move(fut.GetValue()));
360 } else {
361 context->promise.SetException(fut.GetException());
362 }
363 }
364
365 return MakeReadyFuture<>();
366 });
367 }
368 return future;
369 }

382 Promise<size_t, FutureValueType> promise;

395 for (auto it = first; it != last; ++it) {
396 uint32_t idx = it - first;
