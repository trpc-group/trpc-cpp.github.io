22template <class... Ts>
23Future<Ts...> BlockingGet(Future<Ts...>&& f) {
24 detail::Event ev;
25 Future<Ts...> future;
26
27 // Once the `future` is satisfied, our continuation will move the
28 // result into callback funciton and to execute it.
29 std::move(f).Then([&](Future<Ts...>&& fut) {
30 future = std::move(fut);
31 ev.Set();
32 return MakeReadyFuture<>();
33 });
34 // Block until our continuation wakes us up.
35 ev.Wait();
36 return future;
37}

45template <class... Ts>
46std::optional<Future<Ts...>> BlockingTryGet(Future<Ts...>&& future, uint32_t timeout_ms) {
47 struct State {
48 detail::OneshotTimedEvent event;
49
50 // Protects `receiver`.
51 //
52 // Unlike `BlockingGet`, here it's possible that after `event.Wait()` times
53 // out, concurrently the future is satisfied. In this case the continuation
54 // of the future will be racing with us on `receiver`.
55 std::mutex lock;
56 std::optional<Future<Ts...>> receiver;
57
58 explicit State(std::chrono::steady_clock::time_point expires_at)
59 : event(expires_at) {}
60 };
61 auto state = std::make_shared<State>(ReadSteadyClock() + std::chrono::milliseconds(timeout_ms));
62
63 // `state` must be copied here, in case of timeout, we'll leave the scope
64 // before continuation is fired.
65 std::move(future).Then([state](Future<Ts...>&& fut) {
66 std::scoped_lock _(state->lock);
67 state->receiver.emplace(std::move(fut));
68 state->event.Set();
69 return MakeReadyFuture<>();
70 });
71
72 state->event.Wait();
73
74 std::scoped_lock _(state->lock);
75 return state->receiver ? std::move(state->receiver) : std::nullopt;
76}
